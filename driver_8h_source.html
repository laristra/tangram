93  void set_volume_fractions(std::vector<int> const& cell_num_mats,
94  std::vector<int> const& cell_mat_ids,
95  std::vector<double> const& cell_mat_volfracs,
96  std::vector<Point<Dim>>
97  const& cell_mat_centroids = {}) {
98  int nc = mesh_.num_entities(Tangram::Entity_kind::CELL);
99  assert(cell_num_mats.size() == nc);
100 
101  cell_num_mats_ = cell_num_mats;
102  cell_mat_ids_.clear();
103  cell_mat_volfracs_.clear();
104  cell_mat_centroids_.clear();
105 
106  // ims_tols_[0] are tolerances for 0-th moments,
107  // i.e. ims_tols_[0].fun_eps is the tolerance on absolute value of volume
108  double volume_tol = ims_tols_[0].fun_eps;
109 
110  bool insufficient_vol_tol = false;
111  // We only add materials with volumes above the tolerance
112  int offset = 0;
113  for (int icell = 0; icell < nc; icell++) {
114  double cell_volume = mesh_.cell_volume(icell);
115  int ncmats = cell_num_mats_[icell];
116 
117  for (int icmat = 0; icmat < ncmats; icmat++) {
118  double mat_volume = cell_volume*cell_mat_volfracs[offset + icmat];
119  if (mat_volume >= volume_tol) {
120  cell_mat_ids_.push_back(cell_mat_ids[offset + icmat]);
121  cell_mat_volfracs_.push_back(cell_mat_volfracs[offset + icmat]);
122  if (!cell_mat_centroids.empty())
123  cell_mat_centroids_.push_back(cell_mat_centroids[offset + icmat]);
124 
125  // If specified volume tolerance is not too small, ensure that the volume
126  // of the reconstructed material poly will not drop below the tolerance
127  if (mat_volume <= 2*volume_tol)
128  insufficient_vol_tol = true;
129  }
130  else
131  cell_num_mats_[icell]--;
132  }
133  offset += ncmats;
134  }
135 
136  if (insufficient_vol_tol) {
137  std::vector<IterativeMethodTolerances_t>& ims_tols_r =
138  const_cast <std::vector<IterativeMethodTolerances_t>&> (ims_tols_);
139  ims_tols_r[0].fun_eps /= 2;
140  }
141  }

146  void reconstruct(Wonton::Executor_type const *executor = nullptr) {
147 
148  bool distributed = false;
149  int comm_rank = 0;
150  int world_size = 1;
151 
152 #ifdef TANGRAM_ENABLE_MPI
153  MPI_Comm mycomm = MPI_COMM_NULL;
154  auto mpiexecutor = dynamic_cast<Wonton::MPIExecutor_type const *>(executor);
155  if (mpiexecutor && mpiexecutor->mpicomm != MPI_COMM_NULL) {
156  mycomm = mpiexecutor->mpicomm;
157  MPI_Comm_rank(mycomm, &comm_rank);
158  MPI_Comm_size(mycomm, &world_size);
159  if (world_size > 1)
160  distributed = true;
161  }
162 #endif
163 
164  assert(!cell_num_mats_.empty());
165  int ncells = mesh_.num_entities(Tangram::Entity_kind::CELL);
166 
167  if (comm_rank == 0) std::printf("in Driver::run()...\n");
168  {
169  // Instantiate the interface reconstructor class that will
170  // compute the interfaces and compute the pure material submesh
171  // in each cell
172  CellInterfaceReconstructor<Mesh_Wrapper, Dim, MatPoly_Splitter, MatPoly_Clipper>
173  reconstructor(mesh_, ims_tols_, all_convex_);
174 
175  // Tell the reconstructor what materials are in each cell and
176  // what their volume fractions are
177  reconstructor.set_volume_fractions(cell_num_mats_, cell_mat_ids_,
178  cell_mat_volfracs_, cell_mat_centroids_);
179 
180  float tot_seconds = 0.0, xmat_cells_seconds = 0.0;
181  struct timeval begin_timeval, end_timeval, diff_timeval,
182  xmat_begin_timeval, xmat_end_timeval, xmat_diff_timeval;
183 
184  gettimeofday(&begin_timeval, 0);
185 
186  //Normally, we only need CellMatPoly's for multi-material cells,
187  //so we first find their indices and group MMC's based on the number
188  //of contained materials.
189  //Because we only store indices of MMC's, iMMCs[0] vector corresponds
190  //to two-material cells, and iMMCs[i] vector corresponds to MMC's with
191  //(i+2) materials. If the partition contains MMC's with up to n_max
192  //materials, the size of iMMCs vector is therefore (n_max-1).
193  std::vector<std::vector<int>> iMMCs;
194  for (int icell = 0; icell < ncells; icell++) {
195  int nmats = cell_num_mats_[icell];
196  if (nmats < 2)
197  continue;
198  else if (nmats - 1 > iMMCs.size())
199  iMMCs.resize(nmats - 1);
200  iMMCs[nmats - 2].push_back(icell);
201  }
202  cellmatpolys_.resize(ncells);
203 
204  //Reconstructor is set to operate on multi-material cells only.
205  //To improve load balancing, we operate on the cells with the same
206  //number of materials at a time
207  for (int inm = 0; inm < iMMCs.size(); inm++) {
208  int nMMCs = iMMCs[inm].size();
209  if (nMMCs == 0)
210  continue;
211 
212  if (world_size == 1)
213  gettimeofday(&xmat_begin_timeval, 0);
214 
215  reconstructor.set_cell_indices_to_operate_on(iMMCs[inm]);
216 
217  //If reconstruction is performed for a single cell, we do not use transform:
218  //this allows to use transform inside the reconstructor (e.g. to run
219  //multiple threads for material order permutations in MOF)
220  if (nMMCs == 1)
221  cellmatpolys_[iMMCs[inm][0]] = reconstructor(0);
222  else {
223  Tangram::vector<std::shared_ptr<CellMatPoly<Dim>>> MMCs_cellmatpolys(nMMCs);
224 
225  Tangram::transform(make_counting_iterator(0),
226  make_counting_iterator(nMMCs),
227  MMCs_cellmatpolys.begin(), reconstructor);
228  for (int immc = 0; immc < nMMCs; immc++)
229  cellmatpolys_[iMMCs[inm][immc]] = MMCs_cellmatpolys[immc];
230  }
231 
232  if (world_size == 1) {
233  gettimeofday(&xmat_end_timeval, 0);
234  timersub(&xmat_end_timeval, &xmat_begin_timeval, &xmat_diff_timeval);
235  xmat_cells_seconds = xmat_diff_timeval.tv_sec + 1.0E-6*xmat_diff_timeval.tv_usec;
236  std::cout << "Transform Time for " << nMMCs << " " <<
237  inm + 2 << "-material cells (s): " <<
238  xmat_cells_seconds << ", mean time per cell (s): " <<
239  xmat_cells_seconds/nMMCs << std::endl;
240  }
241  }
242  gettimeofday(&end_timeval, 0);
243  timersub(&end_timeval, &begin_timeval, &diff_timeval);
244  tot_seconds = diff_timeval.tv_sec + 1.0E-6*diff_timeval.tv_usec;
245 
246  float max_transform_time = tot_seconds;
247 #ifdef TANGRAM_ENABLE_MPI
248  if (world_size > 1) {
249  MPI_Allreduce(&tot_seconds, &max_transform_time, 1, MPI_FLOAT, MPI_MAX,
250  mycomm);
251  }
252 #endif
253  if (comm_rank == 0)
254  std::cout << "Max Transform Time over " << world_size << " Ranks (s): " <<
255  max_transform_time << std::endl;
256  }
257 
258  reconstruction_done_ = true;
259  }
