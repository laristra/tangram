40 Tangram::Matrix rotation_matrix(const Tangram::Vector3& axis,
41  double rotation_angle) {
42  Tangram::Matrix rot_matrix(3, 3);
43  double cosang = cos(rotation_angle);
44  double sinang = sin(rotation_angle);
45 
46  rot_matrix[0][0] = cosang + axis[0]*axis[0]*(1.0 - cosang);
47  rot_matrix[0][1] = -axis[2]*sinang + axis[0]*axis[1]*(1.0 - cosang);
48  rot_matrix[0][2] = axis[1]*sinang + axis[0]*axis[2]*(1.0 - cosang);
49 
50  rot_matrix[1][0] = axis[2]*sinang + axis[0]*axis[1]*(1.0 - cosang);
51  rot_matrix[1][1] = cosang + axis[1]*axis[1]*(1.0 - cosang);
52  rot_matrix[1][2] = -axis[0]*sinang + axis[1]*axis[2]*(1.0 - cosang);
53 
54  rot_matrix[2][0] = -axis[1]*sinang + axis[0]*axis[2]*(1.0 - cosang);
55  rot_matrix[2][1] = axis[0]*sinang + axis[1]*axis[2]*(1.0 - cosang);
56  rot_matrix[2][2] = cosang + axis[2]*axis[2]*(1.0 - cosang);
57  return rot_matrix;
58 }

70 std::vector<Tangram::Point3> circle3d(const Tangram::Point3& center,
71  double radius,
72  const Tangram::Vector3& normal,
73  int nsamples,
74  double dst_tol) {
75  //We start with the i vector aligned with the x-axis and
76  //the j vector aligned with the x-axis, then pick the one with
77  //the larger projection
78  Tangram::Vector<3> test_vec[2];
79  for (int iv = 0; iv < 2; iv++) {
80  test_vec[iv].axis(iv);
81  //We project the current vector on the circle's plane
82  test_vec[iv] = test_vec[iv] - Wonton::dot(test_vec[iv], normal)*normal;
83  }
84  Tangram::Vector<3> start_vec = (test_vec[0].norm() >= test_vec[1].norm()) ?
85  test_vec[0] : test_vec[1];
86 
87  //Scale to radius length
88  start_vec *= (radius/start_vec.norm());
89 
90  std::vector<Tangram::Point3> circle_pts;
91  circle_pts.reserve(nsamples);
92  //Add the starting point
93  circle_pts.push_back(center + start_vec);
94  //Get the remaining points by rotation
95  for (int isample = 0; isample < nsamples - 1; isample++) {
96  double rot_angle = 2*PI*(isample + 1)/nsamples;
97  Tangram::Matrix rot_matrix = rotation_matrix(normal, rot_angle);
98 
99  circle_pts.push_back(center + rot_matrix*start_vec);
100  }
101 
102  return circle_pts;
103 }

116 std::vector<Tangram::Point3> cog3d(const Tangram::Point3& center,
117  double inner_radius,
118  double outer_radius,
119  const Tangram::Vector3& normal,
120  int nteeth,
121  double dst_tol) {
122  assert(nteeth > 2);
123 
124  //We start with the i vector aligned with the x-axis and
125  //the j vector aligned with the x-axis, then pick the one with
126  //the larger projection
127  Tangram::Vector<3> test_vec[2];
128  for (int iv = 0; iv < 2; iv++) {
129  test_vec[iv].axis(iv);
130  //We project the current vector on the circle's plane
131  test_vec[iv] = test_vec[iv] - Wonton::dot(test_vec[iv], normal)*normal;
132  }
133  Tangram::Vector<3> outer_start_vec = (test_vec[0].norm() >= test_vec[1].norm()) ?
134  test_vec[0] : test_vec[1];
135 
136  double vnorm = outer_start_vec.norm();
137  Tangram::Vector<3> inner_start_vec = outer_start_vec;
138  //Scale to the espective radius length
139  outer_start_vec *= (outer_radius/vnorm);
140  inner_start_vec *= (inner_radius/vnorm);
141 
142  std::vector<Tangram::Point3> cog_pts;
143  cog_pts.reserve(4*nteeth);
144 
145  //We want chords on the inner and the outer circles to be of the same length,
146  //and the segments between circles to be of the same length
147  double tooth_angle = 2*PI/nteeth;
148  double inner_angle = 2*std::atan(
149  sin(tooth_angle/4.0)/(inner_radius/outer_radius + cos(tooth_angle/4.0)) );
150  double dtooth_ang[4] = { 0.0, tooth_angle/2.0 - inner_angle,
151  tooth_angle/4.0, inner_angle };
152 
153  //Get the points by rotation
154  for (int itooth = 0; itooth < nteeth; itooth++) {
155  double rot_angle = 2*PI*itooth/nteeth;
156  for (int its = 0; its < 4; its++) {
157  rot_angle += dtooth_ang[its];
158  Tangram::Matrix rot_matrix = rotation_matrix(normal, rot_angle);
159 
160  if (its < 2)
161  cog_pts.push_back(center + rot_matrix*outer_start_vec);
162  else
163  cog_pts.push_back(center + rot_matrix*inner_start_vec);
164  }
165  }
166 
167  return cog_pts;
168 }

182 Tangram::MatPoly<3> prism(const std::vector<Tangram::Point3>& base_pts,
183  double height,
184  double base_scaling,
185  double dst_tol,
186  const Tangram::Vector3& base_normal =
187  Tangram::Vector3(0.0, 0.0, 0.0)) {
188  int nbase_vrts = (int) base_pts.size();
189  std::vector< std::vector<int> > ifaces_vrts(nbase_vrts + 2);
190  ifaces_vrts[0].resize(nbase_vrts);
191  std::iota(ifaces_vrts[0].begin(), ifaces_vrts[0].end(), 0);
192 
193  std::vector<Tangram::Point3> prism_pts;
194  prism_pts.reserve(2*nbase_vrts);
195  prism_pts.insert(prism_pts.begin(), base_pts.begin(), base_pts.end());
196 
197  Tangram::Vector3 base_shift;
198  if (base_normal.is_zero(dst_tol))
199  base_shift = Tangram::polygon3d_normal(prism_pts, ifaces_vrts[0], dst_tol);
200  else
201  base_shift = base_normal;
202 
203  base_shift *= -height;
204 
205  for(int ivrt = 0; ivrt < nbase_vrts; ivrt++)
206  prism_pts.push_back(Tangram::Point3(
207  prism_pts[(nbase_vrts - ivrt)%nbase_vrts] + base_shift));
208 
209  for(int ivrt = 0; ivrt < nbase_vrts; ivrt++) {
210  int iface = ivrt + 1;
211  ifaces_vrts[iface] = { (ivrt + 1)%nbase_vrts, ivrt,
212  nbase_vrts + (nbase_vrts - ivrt)%nbase_vrts,
213  2*nbase_vrts - ivrt - 1 };
214  }
215 
216  ifaces_vrts[nbase_vrts + 1].resize(nbase_vrts);
217  std::iota(ifaces_vrts[nbase_vrts + 1].begin(),
218  ifaces_vrts[nbase_vrts + 1].end(), nbase_vrts);
219 
220  if (base_scaling != 1.0) {
221  std::vector<double> new_base_moments;
222  Tangram::polygon3d_moments(prism_pts, ifaces_vrts[nbase_vrts + 1],
223  new_base_moments, dst_tol);
224 
225  Tangram::Point3 new_base_cen;
226  for (int idim = 0; idim < 3; idim++)
227  new_base_cen[idim] = new_base_moments[idim + 1]/new_base_moments[0];
228 
229  for(int ivrt = 0; ivrt < nbase_vrts; ivrt++) {
230  Tangram::Vector3 cen2vrt = prism_pts[nbase_vrts + ivrt] - new_base_cen;
231  cen2vrt *= base_scaling;
232  prism_pts[nbase_vrts + ivrt] = new_base_cen + cen2vrt;
233  }
234  }
235 
236  Tangram::MatPoly<3> mat_poly;
237  mat_poly.initialize(prism_pts, ifaces_vrts, dst_tol);
238 
239  return mat_poly;
240 }

255 Tangram::MatPoly<3> skewed_prism(const std::vector<Tangram::Point3>& base_pts,
256  const Tangram::Vector3& opp_base_normal,
257  const Tangram::Point3& opp_base_centroid,
258  double base_scaling,
259  double dst_tol,
260  const Tangram::Vector3& base_normal =
261  Tangram::Vector3(0.0, 0.0, 0.0)) {
262  int nbase_vrts = (int) base_pts.size();
263  std::vector< std::vector<int> > ifaces_vrts(nbase_vrts + 2);
264  ifaces_vrts[0].resize(nbase_vrts);
265  std::iota(ifaces_vrts[0].begin(), ifaces_vrts[0].end(), 0);
266 
267  std::vector<Tangram::Point3> prism_pts;
268  prism_pts.reserve(2*nbase_vrts);
269  prism_pts.insert(prism_pts.begin(), base_pts.begin(), base_pts.end());
270 
271  std::vector<double> base_moments;
272  Tangram::polygon3d_moments(prism_pts, ifaces_vrts[0], base_moments, dst_tol);
273 
274  Tangram::Point3 base_centroid;
275  for (int idim = 0; idim < 3; idim++)
276  base_centroid[idim] = base_moments[idim + 1]/base_moments[0];
277 
278  Tangram::Vector3 base_start_vec = prism_pts[0] - base_centroid;
279  double vnorm = base_start_vec.norm();
280  assert(vnorm >= dst_tol);
281  base_start_vec /= vnorm;
282 
283  //We find the starting point on the opposite plane by intersecing it
284  //with the line parallel to the line conneting centroids
285  Tangram::Vector3 line_dvec = opp_base_centroid - base_centroid;
286  vnorm = line_dvec.norm();
287  assert(vnorm >= dst_tol);
288  line_dvec /= vnorm;
289 
290  double denom = Wonton::dot(line_dvec, opp_base_normal);
291  assert(denom >= dst_tol);
292 
293  Tangram::Vector3 opp_base_start_vec = prism_pts[0] - opp_base_centroid +
294  (Wonton::dot(opp_base_centroid - prism_pts[0], opp_base_normal)/denom)*line_dvec;
295 
296  vnorm = opp_base_start_vec.norm();
297  assert(vnorm >= dst_tol);
298  opp_base_start_vec /= vnorm;
299 
300  //Base normal to check the rotation angle sign
301  Tangram::Vector3 base_normal_vec;
302  if (base_normal.is_zero(dst_tol))
303  base_normal_vec = Tangram::polygon3d_normal(prism_pts, ifaces_vrts[0], dst_tol);
304  else
305  base_normal_vec = base_normal;
306 
307  //We find opposite base vertices by rotating them by the same angle
308  for(int ivrt = 0; ivrt < nbase_vrts; ivrt++) {
309  int icur_vrt = (nbase_vrts - ivrt)%nbase_vrts;
310 
311  Tangram::Vector3 base_cen2vrt_vec = prism_pts[icur_vrt] - base_centroid;
312  double base_cen2vrt_dst = base_cen2vrt_vec.norm();
313  base_cen2vrt_vec /= base_cen2vrt_dst;
314 
315  //acos gives the value from 0 to PI, need sin to determine the angle sign
316  int sign_rot_angle = Wonton::dot(Wonton::cross(base_start_vec, base_cen2vrt_vec),
317  base_normal_vec) > 0.0 ? 1 : -1;
318 
319  double cos_ang = Wonton::dot(base_cen2vrt_vec, base_start_vec);
320  //Due to numerical errors, the value can go out of the cos range
321  if (cos_ang > 1.0 - std::numeric_limits<double>::epsilon()) cos_ang = 1.0;
322  if (cos_ang < -1.0 + std::numeric_limits<double>::epsilon()) cos_ang = -1.0;
323 
324  double rot_angle = sign_rot_angle*std::acos(cos_ang);
325 
326  //Normals are outward, so we flip the direction of rotation
327  Tangram::Matrix rot_matrix = rotation_matrix(opp_base_normal, -rot_angle);
328 
329  //Scale and get vertex coordinate
330  prism_pts.push_back(opp_base_centroid +
331  base_scaling*base_cen2vrt_dst*(rot_matrix*opp_base_start_vec));
332  }
333 
334  for(int ivrt = 0; ivrt < nbase_vrts; ivrt++) {
335  int iface = ivrt + 1;
336  ifaces_vrts[iface] = { (ivrt + 1)%nbase_vrts, ivrt,
337  nbase_vrts + (nbase_vrts - ivrt)%nbase_vrts,
338  2*nbase_vrts - ivrt - 1 };
339  }
340 
341  ifaces_vrts[nbase_vrts + 1].resize(nbase_vrts);
342  std::iota(ifaces_vrts[nbase_vrts + 1].begin(),
343  ifaces_vrts[nbase_vrts + 1].end(), nbase_vrts);
344 
345  std::vector<double> prj_poly_moments;
346  Tangram::polygon3d_moments(prism_pts, ifaces_vrts[nbase_vrts + 1],
347  prj_poly_moments, dst_tol);
348 
349  Tangram::MatPoly<3> mat_poly;
350  mat_poly.initialize(prism_pts, ifaces_vrts, dst_tol);
351 
352  return mat_poly;
353 }

365 Tangram::MatPoly<3> sphere(const Tangram::Point3& center,
366  double radius,
367  int nquadrant_samples,
368  double dst_tol) {
369  std::vector<Tangram::Point3> sphere_pts;
370  int ncircle_samples = 4*nquadrant_samples;
371  sphere_pts.reserve((2*nquadrant_samples - 1)*ncircle_samples + 2);
372 
373  //Top point
374  sphere_pts.push_back(Tangram::Point3(center[0], center[1], center[2] + radius));
375  //Level circles
376  for (int iin = 0; iin < 2*nquadrant_samples - 1; iin++) {
377  double inclination = (iin + 1)*PI/(2*nquadrant_samples);
378  for (int iaz = 0; iaz < ncircle_samples; iaz++) {
379  double azimuth = iaz*PI/(2*nquadrant_samples);
380  sphere_pts.push_back(Tangram::Point3(
381  center[0] + radius*sin(inclination)*cos(azimuth),
382  center[1] + radius*sin(inclination)*sin(azimuth),
383  center[2] + radius*cos(inclination)));
384  }
385  }
386  //Bottom point
387  sphere_pts.push_back(Tangram::Point3(center[0], center[1], center[2] - radius));
388 
389  std::vector< std::vector<int> > ifaces_vrts(8*pow2(nquadrant_samples));
390  //Top triangular faces
391  for (int iaz = 0; iaz < ncircle_samples; iaz++)
392  ifaces_vrts[iaz] = {0, 1 + iaz, 1 + (iaz + 1)%ncircle_samples};
393 
394  //Quad faces
395  for (int iin = 0; iin < 2*(nquadrant_samples - 1); iin++) {
396  int offset = (iin + 1)*ncircle_samples;
397  for (int iaz = 0; iaz < ncircle_samples; iaz++)
398  ifaces_vrts[offset + iaz] = {
399  offset - ncircle_samples + 1 + (iaz + 1)%ncircle_samples,
400  offset - ncircle_samples + 1 + iaz,
401  offset + 1 + iaz,
402  offset + 1 + (iaz + 1)%ncircle_samples };
403  }
404 
405  //Bottom triangular faces
406  int offset = (2*nquadrant_samples - 1)*ncircle_samples;
407  for (int iaz = 0; iaz < ncircle_samples; iaz++)
408  ifaces_vrts[offset + iaz] = {offset + 1,
409  offset - ncircle_samples + 1 + (iaz + 1)%ncircle_samples,
410  offset - ncircle_samples + 1 + iaz };
411 
412  Tangram::MatPoly<3> mat_poly;
413  mat_poly.initialize(sphere_pts, ifaces_vrts, dst_tol);
414 
415  return mat_poly;
416 }
