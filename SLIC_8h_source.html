99  std::shared_ptr<CellMatPoly<Dim>> operator()(const int cell_op_ID) const {
100  double vol_tol = ims_tols_[0].fun_eps; // Volume tolerance
101  double dst_tol = ims_tols_[0].arg_eps; // Distance tolerance
102 
103  int cellID = icells_to_reconstruct[cell_op_ID];
104  auto numMats = cell_num_mats_[cellID];
105 
106  CellMatPoly<Dim>* cellpoly = new CellMatPoly<Dim>(cellID);
107 
108  auto iStart = cell_mat_offsets_[cellID];
109 
110  //Sets of MatPoly's on two sides of the cutting plane
111  HalfSpaceSets_t<Dim> hs_sets;
112 
113  // For every chopped off material, we split MatPoly's above the plane
114  hs_sets.upper_halfspace_set.matpolys.resize(1);
115  Tangram::cell_get_matpoly(mesh_, cellID,
116  &hs_sets.upper_halfspace_set.matpolys[0], dst_tol);
117 
118  // Just going along x-direction
119  Plane_t<Dim> cutting_plane;
120  for (int i = 0 ; i < Dim; i++)
121  cutting_plane.normal[i] = 0.0;
122  cutting_plane.normal[0] = 1.0;
123 
124  //Create Splitter instance
125  MatPoly_Splitter split_matpolys(hs_sets.upper_halfspace_set.matpolys,
126  cutting_plane, vol_tol, dst_tol, all_convex_);
127 
128  //Create cutting distance solver: if not all cells are convex, we assume that
129  //faces are non-planar
130  CuttingDistanceSolver<Dim, MatPoly_Clipper>
131  solve_cut_dst(hs_sets.upper_halfspace_set.matpolys,
132  cutting_plane.normal, ims_tols_[0], all_convex_);
133 
134  //Cutting from left to right
135  double cell_volume = mesh_.cell_volume(cellID);
136  for (int iMat(0); iMat < numMats; iMat++) {
137  double target_vol = cell_mat_volfracs_[iStart + iMat]*cell_volume;
138  // If the target volume is too small, skip it
139  if (target_vol < vol_tol) continue;
140 
141  MatPolySet_t<Dim>* single_mat_set_ptr;
142 
143  //On the last iteration the remaining part is single-material,
144  //so we don't need to split it
145  if (iMat == numMats - 1)
146  single_mat_set_ptr = &hs_sets.upper_halfspace_set;
147  else {
148  // Find distance to origin for the cutting plane
149  solve_cut_dst.set_target_volume(target_vol);
150  std::vector<double> clip_res = solve_cut_dst();
151  cutting_plane.dist2origin = clip_res[0];
152 
153 #ifdef DEBUG
154  // Check if the resulting volume matches the reference value
155  double cur_vol_err = std::fabs(clip_res[1] - target_vol);
156  if (cur_vol_err > vol_tol)
157  std::cerr << "SLIC for cell " << cellID << ": given a maximum of " << ims_tols_[0].max_num_iter <<
158  " iteration(s) achieved error in volume for material " <<
159  cell_mat_ids_[iStart + iMat] << " is " << cur_vol_err <<
160  ", volume tolerance is " << vol_tol << std::endl;
161 #endif
162  hs_sets = split_matpolys();
163 
164  //Chopped off single-material MatPoly's are below the plane
165  single_mat_set_ptr = &hs_sets.lower_halfspace_set;
166  }
167  //Add single-material MatPoly's to CellMatPoly
168  for (int ismp = 0; ismp < single_mat_set_ptr->matpolys.size(); ismp++) {
169  MatPoly<Dim>& cur_matpoly = single_mat_set_ptr->matpolys[ismp];
170  cur_matpoly.set_mat_id(cell_mat_ids_[iStart+iMat]);
171  cellpoly->add_matpoly(cur_matpoly);
172  }
173  }
174 
175  return std::shared_ptr<CellMatPoly<Dim>>(cellpoly);
176  }
