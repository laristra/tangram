99  std::shared_ptr<CellMatPoly<Dim>> operator()(const int cell_op_ID) const {
100  double vol_tol = ims_tols_[0].fun_eps; // Volume tolerance
101 
102  int cellID = icells_to_reconstruct[cell_op_ID];
103  auto numMats = cell_num_mats_[cellID];
104 
105  CellMatPoly<Dim>* cellpoly = new CellMatPoly<Dim>(cellID);
106 
107  auto iStart = cell_mat_offsets_[cellID];
108 
109  //Sets of MatPoly's on two sides of the cutting plane
110  HalfSpaceSets_t<Dim> hs_sets;
111 
112  // For every chopped off material, we split MatPoly's above the plane
113  hs_sets.upper_halfspace_set.matpolys.resize(1);
114  Tangram::cell_get_matpoly(mesh_, cellID, &hs_sets.upper_halfspace_set.matpolys[0]);
115 
116  // Just going along x-direction
117  Plane_t<Dim> cutting_plane;
118  for (int i = 0 ; i < Dim; i++)
119  cutting_plane.normal[i] = 0.0;
120  cutting_plane.normal[0] = 1.0;
121 
122  //Create Splitter instance
123  MatPoly_Splitter split_matpolys(hs_sets.upper_halfspace_set.matpolys,
124  cutting_plane, all_convex_);
125 
126  //Create cutting distance solver: if not all cells are convex, we assume that
127  //faces are non-planar
128  CuttingDistanceSolver<Dim, MatPoly_Clipper>
129  solve_cut_dst(hs_sets.upper_halfspace_set.matpolys,
130  cutting_plane.normal, ims_tols_[0], all_convex_);
131 
132  //Cutting from left to right
133  double cell_volume = mesh_.cell_volume(cellID);
134  for (int iMat(0); iMat < numMats; iMat++) {
135  double target_vol = cell_mat_volfracs_[iStart + iMat]*cell_volume;
136  // If the target volume is too small, skip it
137  if (target_vol < vol_tol) continue;
138 
139  MatPolySet_t<Dim>* single_mat_set_ptr;
140 
141  //On the last iteration the remaining part is single-material,
142  //so we don't need to split it
143  if (iMat == numMats - 1)
144  single_mat_set_ptr = &hs_sets.upper_halfspace_set;
145  else {
146  // Find distance to origin for the cutting plane
147  solve_cut_dst.set_target_volume(target_vol);
148  std::vector<double> clip_res = solve_cut_dst();
149  cutting_plane.dist2origin = clip_res[0];
150 
151 #ifdef DEBUG
152  // Check if the resulting volume matches the reference value
153  double cur_vol_err = std::fabs(clip_res[1] - target_vol);
154  if (cur_vol_err > vol_tol)
155  std::cerr << "SLIC for cell " << cellID << ": after " << ims_tols_[0].max_num_iter <<
156  " iteration(s) achieved error in volume for material " <<
157  cell_mat_ids_[iStart + iMat] << " is " << cur_vol_err <<
158  ", volume tolerance is " << vol_tol << std::endl;
159 #endif
160  hs_sets = split_matpolys();
161 
162  //Chopped off single-material MatPoly's are below the plane
163  single_mat_set_ptr = &hs_sets.lower_halfspace_set;
164  }
165  //Add single-material MatPoly's to CellMatPoly
166  for (int ismp = 0; ismp < single_mat_set_ptr->matpolys.size(); ismp++) {
167  MatPoly<Dim>& cur_matpoly = single_mat_set_ptr->matpolys[ismp];
168  cur_matpoly.set_mat_id(cell_mat_ids_[iStart+iMat]);
169  cellpoly->add_matpoly(cur_matpoly);
170  }
171  }
172 
173  return std::shared_ptr<CellMatPoly<Dim>>(cellpoly);
174  }
