19 double polygon3d_area(const std::vector<Point3>& points,
20  const std::vector<int>& poly_vertices,
21  double dst_tol) {
22  double poly_area = 0.0;
23 
24  int nvrts = static_cast<int>(poly_vertices.size());
25  int nnz_edges = 0;
26  if (nvrts == 3) {
27  Vector<3> side_vec[2];
28  for (int iside = 0; iside < 2; iside++) {
29  side_vec[iside] = points[poly_vertices[iside + 1]] - points[poly_vertices[0]];
30  if (side_vec[iside].norm() >= dst_tol) nnz_edges++;
31  }
32  if ((nnz_edges == 2) &&
33  ((points[poly_vertices[2]] - points[poly_vertices[1]]).norm() >= dst_tol))
34  poly_area = 0.5*cross(side_vec[0], side_vec[1]).norm();
35 
36  return poly_area;
37  }
38 
39  Point3 gcenter;
40  for (int ivrt = 0; ivrt < nvrts; ivrt++)
41  gcenter += points[poly_vertices[ivrt]];
42  gcenter /= nvrts;
43 
44  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
45  int ifv = poly_vertices[ivrt];
46  int isv = poly_vertices[(ivrt + 1)%nvrts];
47  Vector<3> side_vec = points[isv] - points[ifv];
48  if (side_vec.norm() >= dst_tol) nnz_edges++;
49 
50  double tri_size =
51  0.5*cross(side_vec, gcenter - points[ifv]).norm();
52 
53  poly_area += tri_size;
54  }
55 
56  if (nnz_edges < 3)
57  poly_area = 0.0;
58 
59  return poly_area;
60 }

62 void polygon3d_moments(const std::vector<Point3>& points,
63  const std::vector<int>& poly_vertices,
64  std::vector<double>& poly_moments,
65  double dst_tol) {
66  poly_moments.assign(4, 0.0);
67 
68  int nvrts = static_cast<int>(poly_vertices.size());
69  int nnz_edges = 0;
70  if (nvrts == 3) {
71  Vector<3> side_vec[2];
72  for (int iside = 0; iside < 2; iside++) {
73  side_vec[iside] = points[poly_vertices[iside + 1]] - points[poly_vertices[0]];
74  if (side_vec[iside].norm() >= dst_tol) nnz_edges++;
75  }
76 
77  if ((nnz_edges == 2) &&
78  ((points[poly_vertices[2]] - points[poly_vertices[1]]).norm() >= dst_tol)) {
79  poly_moments[0] = 0.5*cross(side_vec[0], side_vec[1]).norm();
80 
81  for (int ivrt = 0; ivrt < 3; ivrt++)
82  for (int idim = 0; idim < 3; idim++)
83  poly_moments[idim + 1] += points[poly_vertices[ivrt]][idim];
84  for (int idim = 0; idim < 3; idim++)
85  poly_moments[idim + 1] *= poly_moments[0]/3.0;
86  }
87 
88  return;
89  }
90 
91  Point3 gcenter;
92  for (int ivrt = 0; ivrt < nvrts; ivrt++)
93  gcenter += points[poly_vertices[ivrt]];
94  gcenter /= nvrts;
95 
96  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
97  int ifv = poly_vertices[ivrt];
98  int isv = poly_vertices[(ivrt + 1)%nvrts];
99  Vector<3> side_vec = points[isv] - points[ifv];
100  if (side_vec.norm() >= dst_tol) nnz_edges++;
101 
102  double tri_size =
103  0.5*cross(side_vec, gcenter - points[ifv]).norm();
104 
105  poly_moments[0] += tri_size;
106  for (int idim = 0; idim < 3; idim++)
107  poly_moments[idim + 1] +=
108  tri_size*(points[ifv][idim] + points[isv][idim] + gcenter[idim])/3.0;
109  }
110 
111  if (nnz_edges < 3)
112  poly_moments.assign(4, 0.0);
113 }

115 Vector<3> polygon3d_normal(const std::vector<Point3>& points,
116  const std::vector<int>& poly_vertices,
117  double dst_tol) {
118  Vector3 poly_normal(0.0, 0.0, 0.0);
119 
120  int nvrts = static_cast<int>(poly_vertices.size());
121  if (nvrts == 3) {
122  int nnz_edges = 0;
123  Vector<3> side_vec[2];
124  for (int iside = 0; iside < 2; iside++) {
125  side_vec[iside] = points[poly_vertices[iside + 1]] - points[poly_vertices[0]];
126  if (side_vec[iside].norm() >= dst_tol) nnz_edges++;
127  }
128 
129  if ((nnz_edges == 2) &&
130  ((points[poly_vertices[2]] - points[poly_vertices[1]]).norm() >= dst_tol)) {
131  poly_normal = cross(side_vec[0], side_vec[1]);
132  poly_normal.normalize();
133  }
134 
135  return poly_normal;
136  }
137 
138  std::vector<double> poly_moments;
139  polygon3d_moments(points, poly_vertices, poly_moments, dst_tol);
140 
141  if (poly_moments[0] == 0.0)
142  return poly_normal;
143 
144  Point3 poly_centroid;
145  for (int idim = 0; idim < 3; idim++)
146  poly_centroid[idim] = poly_moments[idim + 1]/poly_moments[0];
147 
148  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
149  Vector<3> side_vec = points[poly_vertices[(ivrt + 1)%nvrts]] -
150  points[poly_vertices[ivrt]];
151  Vector<3> tri_normal = cross(side_vec, poly_centroid - points[poly_vertices[ivrt]]);
152  poly_normal += tri_normal;
153  }
154  poly_normal.normalize();
155 
156  return poly_normal;
157 }

551 template<>
552 void MatPoly<2>::compute_moments(std::vector<double>& moments) const {
553  moments.assign(3, 0.0);
554 
555  for (int ivrt = 0; ivrt < nvertices_; ivrt++) {
556  double cur_term = vertex_points_[ivrt][0]*vertex_points_[(ivrt + 1)%nvertices_][1] -
557  vertex_points_[ivrt][1]*vertex_points_[(ivrt + 1)%nvertices_][0];
558  moments[0] += cur_term;
559  for (int idim = 0; idim < 2; idim++)
560  moments[idim + 1] += cur_term*(
561  vertex_points_[ivrt][idim] + vertex_points_[(ivrt + 1)%nvertices_][idim]);
562  }
563  moments[0] /= 2.0;
564  for (int idim = 0; idim < 2; idim++)
565  moments[idim + 1] /= 6.0;
566 }

573 template<>
574 void MatPoly<3>::compute_moments(std::vector<double>& moments) const {
575  moments.assign(4, 0.0);
576 
577  for (int iface = 0; iface < nfaces_; iface++) {
578  std::vector<double> face_moments;
579  polygon3d_moments(vertex_points_, face_vertices_[iface], face_moments, dst_tol_);
580 
581  if (face_moments[0] == 0.0)
582  continue;
583 
584  std::vector<Point3> face_pts = face_points(iface);
585  std::vector< std::vector<int> > itri_pts;
586 
587  int nvrts = face_vertices_[iface].size();
588  if (nvrts == 3)
589  itri_pts.push_back({0, 1, 2});
590  else {
591  itri_pts.reserve(nvrts);
592  face_pts.push_back(Point3(face_moments[1]/face_moments[0],
593  face_moments[2]/face_moments[0], face_moments[3]/face_moments[0]));
594  for (int ivrt = 0; ivrt < nvrts; ivrt++)
595  itri_pts.push_back({nvrts, ivrt, (ivrt + 1)%nvrts});
596  }
597 
598  for (int itri = 0; itri < itri_pts.size(); itri++) {
599  Vector3 vcp = cross(face_pts[itri_pts[itri][1]] - face_pts[itri_pts[itri][0]],
600  face_pts[itri_pts[itri][2]] - face_pts[itri_pts[itri][0]]);
601 
602  moments[0] += dot(vcp, face_pts[itri_pts[itri][0]].asV());
603  for (int idim = 0; idim < 3; idim++)
604  for (int ivrt = 0; ivrt < 3; ivrt++)
605  moments[idim + 1] += vcp[idim]*pow(face_pts[itri_pts[itri][ivrt]][idim] +
606  face_pts[itri_pts[itri][(ivrt + 1)%3]][idim], 2);
607  }
608  }
609 
610  moments[0] /= 6.0;
611  for (int idim = 0; idim < 3; idim++)
612  moments[idim + 1] /= 48.0;
613 }

886 MatPoly<2> natural_selection(const std::vector< Point<2> >& poly_points,
887  const double dst_tol,
888  const std::vector< Point<2> >* reference_pts = nullptr) {
889  MatPoly<2> fit_poly;
890  int nvrts = static_cast<int>(poly_points.size());
891  std::vector< Point<2> > unique_points;
892  unique_points.reserve(nvrts);
893 
894  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
895  // Check if this point is coincident with the next one,
896  // i.e. if the respective edge is degenerate: points
897  // are considered coincident if the distance between them
898  // is below the distance tolerance
899  if ((poly_points[ivrt] - poly_points[(ivrt + 1)%nvrts]).norm() >= dst_tol) {
900  bool reference_match = false;
901  if (reference_pts != nullptr) {
902  for (int irp = 0; irp < reference_pts->size(); irp++)
903  if ((poly_points[ivrt] - (*reference_pts)[irp]).norm() < dst_tol) {
904  unique_points.push_back((*reference_pts)[irp]);
905  reference_match = true;
906  break;
907  }
908  }
909 
910  if (!reference_match)
911  unique_points.push_back(poly_points[ivrt]);
912  }
913  }
914  unique_points.shrink_to_fit();
915 
916  // Polygons with less than three vertices are clearly degenerate
917  if (unique_points.size() > 2)
918  fit_poly.initialize(unique_points, dst_tol);
919 
920  return fit_poly;
921 }

934 MatPoly<3> natural_selection(const std::vector<Point3>& vertex_points,
935  const std::vector< std::vector<int> >& face_vertices,
936  const double dst_tol,
937  const std::vector< Point<3> >* reference_pts = nullptr) {
938  MatPoly<3> fit_poly;
939  int nvrts = static_cast<int>(vertex_points.size());
940  std::vector<Point3> unique_vrts;
941  unique_vrts.reserve(nvrts);
942  // We only store unique vertices, so we need a map from the original node indices
943  // to unique node indices for when we process faces
944  std::vector<int> orig2unique_vrt_ids(nvrts, -1);
945  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
946  for (int i = 0; i < unique_vrts.size(); i++) {
947  // Check if this point is coincident with a stored one: points
948  // are considered coincident if the distance between them is below
949  // the distance tolerance
950  if ((vertex_points[ivrt] - unique_vrts[i]).norm() < dst_tol) {
951  orig2unique_vrt_ids[ivrt] = i;
952  break;
953  }
954  }
955  // If the point is unique, we store it
956  if (orig2unique_vrt_ids[ivrt] == -1) {
957  orig2unique_vrt_ids[ivrt] = static_cast<int>(unique_vrts.size());
958  bool reference_match = false;
959  if (reference_pts != nullptr) {
960  // We check if this point matches any of the reference ones
961  for (int irp = 0; irp < reference_pts->size(); irp++) {
962  const Point<3>& cur_ref_pt = (*reference_pts)[irp];
963  if ((vertex_points[ivrt] - cur_ref_pt).norm() < dst_tol) {
964  unique_vrts.push_back(cur_ref_pt);
965  reference_match = true;
966  break;
967  }
968  }
969  }
970  if (!reference_match)
971  unique_vrts.push_back(vertex_points[ivrt]);
972  }
973  }
974  unique_vrts.shrink_to_fit();
975  nvrts = static_cast<int>(unique_vrts.size());
976 
977  // Check if it still has a chance to be at least a tetrahedron
978  if (nvrts < 4) return fit_poly;
979 
980  //First pass on faces: have them reference only unique vertices
981  int nfaces = static_cast<int>(face_vertices.size());
982  std::vector< std::vector<int> > face_unique_vrts(nfaces);
983  for (int iface = 0; iface < nfaces; iface++) {
984  int face_nvrts = static_cast<int>(face_vertices[iface].size());
985  for (int ifv = 0; ifv < face_nvrts; ifv++) {
986  int cur_vrt_id = orig2unique_vrt_ids[face_vertices[iface][ifv]];
987  // There can be only one (c) such index in the list of face indices
988  if (std::find(face_unique_vrts[iface].begin(), face_unique_vrts[iface].end(),
989  cur_vrt_id) == face_unique_vrts[iface].end())
990  face_unique_vrts[iface].push_back(cur_vrt_id);
991  }
992  }
993 
994  // Hanging vertices can be a problem, so we first find how many unique
995  // connections is in each vertice's network
996  std::vector< std::vector<int> > connected_vrt_ids(nvrts);
997  std::vector< std::vector< std::pair<int, int> > > vrt_in_faces(nvrts);
998  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
999  for (int iface = 0; iface < face_unique_vrts.size(); iface++) {
1000  const std::vector<int>& cur_face_vrts = face_unique_vrts[iface];
1001  int nface_vrts = static_cast<int>(cur_face_vrts.size());
1002  int id_in_face = std::distance(cur_face_vrts.begin(),
1003  std::find(cur_face_vrts.begin(),
1004  cur_face_vrts.end(), ivrt));
1005 
1006  if (id_in_face != nface_vrts) {
1007  // Vertex is in the current face
1008  vrt_in_faces[ivrt].push_back(std::pair<int, int>(iface, id_in_face));
1009  // We check if it has unknown connections with vertices of the current face
1010  for (int p0n1 = 0; p0n1 < 2; p0n1++) {
1011  int adj_vrt_face_id = (nface_vrts + id_in_face + 2*p0n1 - 1)%nface_vrts;
1012  int adj_vrt_id = face_unique_vrts[iface][adj_vrt_face_id];
1013  if (std::find(connected_vrt_ids[ivrt].begin(), connected_vrt_ids[ivrt].end(),
1014  adj_vrt_id) == connected_vrt_ids[ivrt].end())
1015  connected_vrt_ids[ivrt].push_back(adj_vrt_id);
1016  }
1017  }
1018  }
1019  }
1020 
1021  // Now we dispose of hanger-on vertices that don't have enough connections
1022  // This also affects the connections of vertices which they networked with,
1023  // so we need to check if those get dragged down the cliff as well
1024  bool dropped_connections;
1025  std::vector<int> hanging_vrts_ids;
1026  do {
1027  dropped_connections = false;
1028  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
1029  int ncv = static_cast<int>(connected_vrt_ids[ivrt].size());
1030  if (ncv < 3) {
1031  hanging_vrts_ids.push_back(ivrt);
1032  // Others break connections first
1033  for (int icv = 0; icv < ncv; icv++) {
1034  int icvrt = connected_vrt_ids[ivrt][icv];
1035  connected_vrt_ids[icvrt].erase(
1036  std::find(connected_vrt_ids[icvrt].begin(),
1037  connected_vrt_ids[icvrt].end(), ivrt));
1038  }
1039  if (ncv == 1) dropped_connections = true;
1040 
1041  bool third_wheel = false;
1042  for (int icf = 0; icf < vrt_in_faces[ivrt].size(); icf++) {
1043  int iface = vrt_in_faces[ivrt][icf].first;
1044  int hvrt_fid = vrt_in_faces[ivrt][icf].second;
1045  // Check if the two others can connect by themselves
1046  if (ncv == 2) {
1047  int nface_vrts = static_cast<int>(face_unique_vrts[iface].size());
1048  for (int icv = 0; icv < 2; icv++)
1049  if ( (face_unique_vrts[iface][(hvrt_fid + 1)%nface_vrts] ==
1050  connected_vrt_ids[ivrt][(icv + 1)%2]) &&
1051  (face_unique_vrts[iface][(nface_vrts + hvrt_fid - 1)%nface_vrts] ==
1052  connected_vrt_ids[ivrt][icv]) ) {
1053  third_wheel = true;
1054  for (int i = 0; i < 2; i++)
1055  connected_vrt_ids[connected_vrt_ids[ivrt][i]].push_back(
1056  connected_vrt_ids[ivrt][(i + 1)%2]);
1057  // And the hanger-on now has no one to cling to
1058  connected_vrt_ids[ivrt].clear();
1059  ncv = 0;
1060  }
1061  }
1062  // Gets evicted from the hosting face
1063  face_unique_vrts[iface].erase(face_unique_vrts[iface].begin() +
1064  vrt_in_faces[ivrt][icf].second);
1065  // Tell the following face vertices about their change of address
1066  for (int ifv = vrt_in_faces[ivrt][icf].second;
1067  ifv < face_unique_vrts[iface].size(); ifv++) {
1068  int icur_vrt = face_unique_vrts[iface][ifv];
1069  for (int i = 0; i < vrt_in_faces[icur_vrt].size(); i++)
1070  if (vrt_in_faces[icur_vrt][i].first == iface) {
1071  vrt_in_faces[icur_vrt][i].second--;
1072  break;
1073  }
1074  }
1075 
1076  }
1077  if (!third_wheel) dropped_connections = true;
1078  }
1079  }
1080  } while (dropped_connections);
1081 
1082  // Filter out degenerate faces
1083  int face_id = 0;
1084  while (face_id < face_unique_vrts.size()) {
1085  // Check if this face is still at least a triangle
1086  if (face_unique_vrts[face_id].size() > 2) face_id++;
1087  else {
1088  //Dispose of the degenerate face
1089  face_unique_vrts.erase(face_unique_vrts.begin() + face_id);
1090  }
1091  }
1092 
1093  // Check if it managed to remain at least a tetrahedron
1094  nfaces = static_cast<int>(face_unique_vrts.size());
1095  if (nfaces < 4) return fit_poly;
1096 
1097  // Only the surviving vertices get remembered
1098  if (!hanging_vrts_ids.empty()) {
1099  std::vector<int> unique2fit_vrt_ids(nvrts, -1);
1100  std::vector<Point3> fit_vrts;
1101  fit_vrts.reserve(nvrts - hanging_vrts_ids.size());
1102  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
1103  // Check if this point is still around
1104  if (std::find(hanging_vrts_ids.begin(), hanging_vrts_ids.end(), ivrt) ==
1105  hanging_vrts_ids.end()) {
1106  unique2fit_vrt_ids[ivrt] = fit_vrts.size();
1107  fit_vrts.push_back(unique_vrts[ivrt]);
1108  }
1109  }
1110  unique_vrts = fit_vrts;
1111 
1112  std::vector< std::vector<int> > pruned_faces(nfaces);
1113  for (int iface = 0; iface < nfaces; iface++) {
1114  int face_nverts = static_cast<int>(face_unique_vrts[iface].size());
1115  pruned_faces[iface].resize(face_nverts);
1116  for (int ifv = 0; ifv < face_nverts; ifv++)
1117  pruned_faces[iface][ifv] = unique2fit_vrt_ids[face_unique_vrts[iface][ifv]];
1118  }
1119  face_unique_vrts = pruned_faces;
1120  }
1121 
1122  // And they lived happily ever after
1123  fit_poly.initialize(unique_vrts, face_unique_vrts, dst_tol);
1124  return fit_poly;
1125 }

1136 bool point_inside_matpoly(const MatPoly<3> mat_poly,
1137  const Point<3>& pt,
1138  double dst_tol,
1139  bool convex_poly) {
1140  std::vector< MatPoly<3> > convex_polys;
1141  if (convex_poly)
1142  convex_polys.push_back(mat_poly);
1143  else
1144  mat_poly.facetize_decompose(convex_polys);
1145 
1146  for (int icp = 0; icp < convex_polys.size(); icp++) {
1147  const std::vector< Point<3> >& poly_pts = convex_polys[icp].points();
1148 
1149  bool pt_inside_cur_poly = true;
1150  for (int iface = 0; iface < convex_polys[icp].num_faces(); iface++) {
1151  const std::vector<int>& iface_vrts = convex_polys[icp].face_vertices(iface);
1152  Vector3 pt2vrt_vec;
1153  double pt2vrt_dst = 0.0;
1154  //Find a face vertex that is the farthest from the given point
1155  for (int ifvrt = 0; ifvrt < iface_vrts.size(); ifvrt++) {
1156  Vector3 cur_pt2vrt_vec =
1157  convex_polys[icp].vertex_point(iface_vrts[ifvrt]) - pt;
1158  double cur_vec_norm = cur_pt2vrt_vec.norm();
1159  if (cur_vec_norm > pt2vrt_dst) {
1160  pt2vrt_vec = cur_pt2vrt_vec;
1161  pt2vrt_dst = cur_vec_norm;
1162  }
1163  }
1164  pt2vrt_vec /= pt2vrt_dst;
1165 
1166  Vector3 face_normal = polygon3d_normal(poly_pts, iface_vrts, dst_tol);
1167  if (face_normal.is_zero(dst_tol))
1168  continue;
1169 
1170  double dist2plane = dot(pt2vrt_vec, face_normal);
1171  //Check the signed distance to the plane
1172  if (dist2plane <= dst_tol) {
1173  pt_inside_cur_poly = false;
1174  break;
1175  }
1176  }
1177 
1178  if (pt_inside_cur_poly)
1179  return true;
1180  }
1181  return false;
1182 }
