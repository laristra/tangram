141  Matrix inverse() const {
142  if (Rows_ != Columns_) {
143  std::cerr << "Matrix is rectangular" << std::endl;
144  throw std::exception();
145  }
146 
147  Matrix Ainv(Rows_, Rows_, 0.0);
148 
149  // Create a temporary matrix with twice as many columns
150  Matrix D(Rows_, 2*Rows_);
151 
152  // Initialize the reduction matrix
153  int Rows2 = 2*Rows_;
154  for (int i = 0; i < Rows_; i++) {
155  for (int j = 0; j < Rows_; j++) {
156  D[i][j] = A_[i*Columns_+j];
157  D[i][Rows_+j] = 0.0;
158  }
159  D[i][Rows_+i] = 1.0;
160  }
161 
162  // Do the reduction
163  for (int i = 0; i < Rows_; i++) {
164  double alpha = D[i][i];
165  if (alpha == 0.0) {
166  std::cerr << "invert: Singular Matrix" << std::endl;
167  return Ainv;
168  }
169 
170  for (int j = 0; j < Rows2; j++)
171  D[i][j] = D[i][j]/alpha;
172 
173  for (int k = 0; k < Rows_; k++) {
174  if ((k - i) == 0)
175  continue;
176 
177  double beta = D[k][i];
178  for (int j = 0; j < Rows2; j++)
179  D[k][j] = D[k][j] - beta*D[i][j];
180  }
181  }
182 
183  // Copy result into output matrix
184  for (int i = 0; i < Rows_; i++)
185  for (int j = 0; j < Rows_; j++)
186  Ainv[i][j] = D[i][j + Rows_];
187 
188  return Ainv;
189  }

333 template<long D>
334 void solve(const Matrix& A, const Vector<D>& b, Vector<D>& x) {
335  int n = A.rows();
336  assert(n == A.columns());
337  assert(D == n);
338 
339  const std::vector<double>& a_ = A.data();
340  double s, norm;
341 
342  Matrix R(A);
343  Matrix Q(n, n, 0.0);
344  Matrix U(n, n);
345 
346  std::vector<double> y(n);
347 
348  int i, j, k, l;
349  for (i = 0; i < n; i++)
350  Q[i][i] = 1.0;
351 
352  // Find A = QR using reflection method
353  for (i = 0; i < n - 1; i++) {
354  y[i] = R[i][i];
355  for (j = i + 1, s = 0.0; j < n; j++) {
356  y[j] = R[j][i];
357  s += pow2(R[j][i]);
358  }
359  norm = sqrt(s + pow2(y[i]));
360  assert(std::fabs(norm > std::numeric_limits<double>::epsilon()));
361  if (s == 0)
362  continue;
363 
364  y[i] -= norm;
365  norm = sqrt(s + pow2(y[i]));
366 
367  for (j = i; j < n; j++)
368  y[j] /= norm;
369 
370  // Generate U matrix
371  for (k = 0; k < n; k++)
372  for (l = 0; l < n; l++)
373  if (k < i || l < i)
374  U[k][l] = (k == l) ? 1.0 : 0.0;
375  else
376  U[k][l] = (k == l) ? 1.0 - 2.0*y[k]*y[l] : -2.0*y[k]*y[l];
377 
378  // Modify Q matrix
379  Q = Q*U;
380  // Modify R matrix
381  R = U*R;
382  }
383  // We need to check last element for non-singularity
384  assert(std::fabs(R[n - 1][n - 1]) > std::numeric_limits<double>::epsilon());
385 
386  Vector<D> QTb = Q.transpose()*b;
387 
388  // Now, use back substitution to find x
389  for (i = n - 1; i >= 0; i--) {
390  for (k = i + 1; k < n; k++)
391  QTb[i] -= R[i][k]*x[k];
392 
393  x[i] = QTb[i]/R[i][i];
394  }
395 }
