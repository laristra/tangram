33 double quad_interpolant_minimizer(const double arg0,
34  const double arg1,
35  const double fval0,
36  const double fval1,
37  const double df0,
38  const double* safeguards) {
39  double darg = arg1 - arg0;
40  double denominator = fval1 - fval0 - df0*(darg);
41  std::vector<double> denominator_terms = {fval0, fval1, df0*(darg)};
42  double max_denominator_term = *std::max(denominator_terms.begin(), denominator_terms.end());
43  double arg_min;
44  double bnd0 = arg0 + safeguards[0]*darg,
45  bnd1 = arg0 + safeguards[1]*darg;
46 
47  if (std::fabs(denominator/max_denominator_term) <=
48  std::numeric_limits<double>::epsilon())
49  arg_min = (fval1 > fval0) ? bnd0 : bnd1;
50  else {
51  arg_min = arg0 - df0*pow2(darg)/(2*denominator);
52  double d2arg0 = std::fabs(arg_min - arg0);
53  if (d2arg0 < std::fabs(bnd0 - arg0)) arg_min = bnd0;
54  else if (d2arg0 > std::fabs(bnd1 - arg0)) arg_min = bnd1;
55  }
56 
57  return arg_min;
58 }

69 template<int arg_dim>
70 void fwd_diff_grad(const std::function<double(const Vector<arg_dim>&)>& fun,
71  const Vector<arg_dim>& arg_val,
72  const double& fun_val,
73  Vector<arg_dim>& approx_grad,
74  double& fdiff_h) {
75  fdiff_h = sqrt(std::numeric_limits<double>::epsilon());
76  Vector<arg_dim> fwd_arg = arg_val;
77  for (int i = 0; i < arg_dim; i++) {
78  fwd_arg[i] += fdiff_h;
79  double forward_fval = fun(fwd_arg);
80  fwd_arg[i] = arg_val[i];
81 
82  approx_grad[i] = (forward_fval - fun_val)/fdiff_h;
83  }
84 }

94 template<int arg_dim>
95 void cen_diff_grad(const std::function<double(const Vector<arg_dim>&)>& fun,
96  const Vector<arg_dim>& arg_val,
97  Vector<arg_dim>& approx_grad,
98  double& fdiff_h) {
99  fdiff_h = pow(std::numeric_limits<double>::epsilon(), 1.0/3.0);
100 
101  Vector<arg_dim> cur_arg = arg_val;
102  for (int i = 0; i < arg_dim; i++) {
103  cur_arg[i] += fdiff_h;
104  double forward_fval = fun(cur_arg);
105  cur_arg[i] = arg_val[i] - fdiff_h;
106  double backward_fval = fun(cur_arg);
107  cur_arg[i] = arg_val[i];
108 
109  approx_grad[i] = (forward_fval - backward_fval)/(2*fdiff_h);
110  }
111 }

125 template<int arg_dim>
126 void finite_diff_grad(const std::function<double(const Vector<arg_dim>&)>& fun,
127  const Vector<arg_dim>& arg_val,
128  const double& fun_val,
129  const double rel_err_bnd,
130  Vector<arg_dim>& approx_grad) {
131  double fdiff_h;
132  fwd_diff_grad(fun, arg_val, fun_val, approx_grad, fdiff_h);
133 
134  double max_norm = approx_grad.max_norm();
135  if (fdiff_h*max_norm/fun_val < std::numeric_limits<double>::epsilon()) {
136  approx_grad.zero();
137  return;
138  }
139  if (fdiff_h/max_norm <= rel_err_bnd) return;
140 
141  //Fallback to central differences
142  cen_diff_grad(fun, arg_val, approx_grad, fdiff_h);
143  if (2*fdiff_h*max_norm/fun_val < std::numeric_limits<double>::epsilon())
144  approx_grad.zero();
145 }

165 template<int arg_dim>
166 double linesearch(const std::function<double(const Vector<arg_dim>&)>& obj_fun,
167  const double& obj_fun_lbnd,
168  const Vector<arg_dim>& arg,
169  const Vector<arg_dim>& dir,
170  const double& initial_guess,
171  const double& c1,
172  const double& c2,
173  const IterativeMethodTolerances_t& im_tols,
174  const double rel_err_bnd,
175  double& cur_fval,
176  Vector<arg_dim>& cur_grad) {
177  // Bracketing step, solution will be in (alpha_bnd[0], alpha_bnd[1])
178  double fval0 = cur_fval;
179  double df0 = dot(cur_grad, dir);
180  assert(df0 < 0.0);
181  double alpha_tol = im_tols.arg_eps/dir.norm();
182  double fdiff_err;
183 
184  auto fval = [&obj_fun, &arg, &dir](double alpha) {
185  return obj_fun(arg + alpha*dir);
186  };
187 
188  // alpha_max is chosen so that it should violate the sufficient decrease condition
189  // and therefore provides an acceptable upper bound for the interval
190  double alpha_max = (obj_fun_lbnd - fval0)/(c1*df0);
191 
192  double alpha_bnd[2] = {0.0, initial_guess};
193  double bnd_fval[2]; bnd_fval[0] = fval0;
194  Vector<arg_dim>& lbnd_grad = cur_grad;
195  double df_lbnd = df0;
196  // At the beginning of each loop, cur_fval and cur_grad are at alpha_bnd[0]
197  for (int i = 0; i < im_tols.max_num_iter; i++) {
198  double tau = 9.0; // interval length scaling
199 
200  bnd_fval[1] = fval(alpha_bnd[1]);
201 
202  if (bnd_fval[1] - obj_fun_lbnd < im_tols.fun_eps) {
203  cur_fval = bnd_fval[1];
204  return alpha_bnd[1];
205  }
206 
207  if ( (bnd_fval[1] > fval0 + c1*alpha_bnd[1]*df0) ||
208  ((bnd_fval[1] >= bnd_fval[0]) && (i > 0)) )
209  break;
210 
211  cur_fval = bnd_fval[1];
212  finite_diff_grad<arg_dim>(obj_fun, arg + alpha_bnd[1]*dir, bnd_fval[1],
213  rel_err_bnd, cur_grad);
214  // alpha_bnd[1] will be the next alpha_bnd[0]
215  df_lbnd = dot(cur_grad, dir);
216 
217  if (std::fabs(df_lbnd) <= -c2*df0)
218  return alpha_bnd[1];
219 
220  lbnd_grad = cur_grad;
221  if (df_lbnd >= 0.0) {
222  std::reverse(std::begin(alpha_bnd), std::end(alpha_bnd));
223  std::reverse(std::begin(bnd_fval), std::end(bnd_fval));
224  break;
225  }
226 
227  double dalpha = alpha_bnd[1] - alpha_bnd[0];
228 
229  alpha_bnd[0] = alpha_bnd[1];
230  bnd_fval[0] = bnd_fval[1];
231  alpha_bnd[1] += tau*dalpha;
232 
233  if (alpha_bnd[1] > alpha_max - alpha_tol) {
234  alpha_bnd[1] = alpha_max;
235  bnd_fval[1] = obj_fun(alpha_max);
236  break;
237  }
238  }
239 
240  // Sectioning/zooming step
241  // At the beginning of each loop, cur_fval and cur_grad are at alpha_bnd[0]
242  double safeguards[2] = {0.1, 0.5};
243  for (int i = 0; i < im_tols.max_num_iter; i++) {
244  if (std::fabs(alpha_bnd[1] - alpha_bnd[0]) < alpha_tol) {
245  cur_grad = lbnd_grad;
246  return alpha_bnd[0];
247  }
248 
249  double cur_alpha = quad_interpolant_minimizer(alpha_bnd[0], alpha_bnd[1], bnd_fval[0],
250  bnd_fval[1], df_lbnd, safeguards);
251  cur_fval = fval(cur_alpha);
252  if (cur_fval - obj_fun_lbnd < im_tols.fun_eps)
253  return cur_alpha;
254 
255  if ( (cur_fval > fval0 + c1*cur_alpha*df0) || (cur_fval >= bnd_fval[0]) ) {
256  alpha_bnd[1] = cur_alpha;
257  bnd_fval[1] = cur_fval;
258  continue;
259  }
260 
261  finite_diff_grad<arg_dim>(obj_fun, arg + cur_alpha*dir, cur_fval,
262  rel_err_bnd, cur_grad);
263 
264  double cur_df = dot(cur_grad, dir);
265  if (std::fabs(cur_df) <= -c2*df0)
266  return cur_alpha;
267 
268  if (cur_df*(alpha_bnd[1] - alpha_bnd[0]) >= 0) {
269  alpha_bnd[1] = alpha_bnd[0];
270  bnd_fval[1] = bnd_fval[0];
271  }
272 
273  alpha_bnd[0] = cur_alpha;
274  bnd_fval[0] = cur_fval;
275  lbnd_grad = cur_grad;
276  df_lbnd = cur_df;
277  }
278 
279  cur_fval = bnd_fval[0];
280  cur_grad = lbnd_grad;
281  return alpha_bnd[0];
282 }

292 template<int arg_dim>
293 Vector<arg_dim> bfgs(const std::function<double(const Vector<arg_dim>&)>& obj_fun,
294  const double& obj_fun_lbnd,
295  const Vector<arg_dim>& init_guess,
296  const IterativeMethodTolerances_t& im_tols) {
297  const double linesearch_c1 = 1.0e-4;
298  const double linesearch_c2 = 0.9;
299  const double damping_scalar = 0.2;
300  const double grad_rel_err_bnd = 0.1;
301 
302  //Initial Hessian approximation: identity matrix
303  Matrix B(arg_dim, arg_dim, 0.0);
304  for (int i = 0; i < arg_dim; i++)
305  B[i][i] = 1.0;
306  Matrix B_old;
307 
308  Vector<arg_dim> cur_arg = init_guess, prev_grad, cur_grad, cur_dir, darg, dgrad;
309  double dfval, cur_fval = obj_fun(cur_arg);
310  finite_diff_grad<arg_dim>(obj_fun, cur_arg, cur_fval,
311  grad_rel_err_bnd, cur_grad);
312 
313  for (int i = 0; i < im_tols.max_num_iter; i++) {
314  // Check if gradient is nonzero, we're finished otherwise
315  if (cur_grad.max_norm() == 0.0)
316  return cur_arg;
317 
318  Wonton::solve<arg_dim>(B, -cur_grad, cur_dir);
319  double df0 = dot(cur_grad, cur_dir);
320 
321  // Function should be decreasing in the linesearch direction
322  if (df0 >= 0.0) {
323  if (i > 0) {
324  B = B_old;
325  Wonton::solve<arg_dim>(B, -cur_grad, cur_dir);
326  df0 = dot(cur_grad, cur_dir);
327  }
328  // Can't decrease the function value any further, finishing
329  if (df0 >= 0.0)
330  return cur_arg;
331  }
332 
333  double initial_alpha = 1.0;
334  if (i > 0) initial_alpha = std::min(1.0,
335  -2*std::max(-dfval, 10*std::numeric_limits<double>::epsilon())/df0);
336 
337  dfval = -cur_fval;
338  prev_grad = cur_grad;
339 
340  double alpha = linesearch<arg_dim>(obj_fun, obj_fun_lbnd,
341  cur_arg, cur_dir, initial_alpha,
342  linesearch_c1, linesearch_c2,
343  im_tols, grad_rel_err_bnd,
344  cur_fval, cur_grad);
345  darg = alpha*cur_dir;
346  // Check if the change in argument was above tolerance, finish otherwise
347  if (darg.norm() < im_tols.arg_eps) return cur_arg;
348 
349  dfval += cur_fval;
350  // Check if the change in function was numerically significant, finish otherwise
351  if (std::fabs(dfval/cur_fval) <= std::numeric_limits<double>::epsilon())
352  return cur_arg;
353 
354  cur_arg += darg;
355  // Check if the value of the function is above tolerance, finish otherwise
356  if (std::fabs(cur_fval) < im_tols.fun_eps) return cur_arg;
357 
358  dgrad = cur_grad - prev_grad;
359 
360  bool valid_dgrad = false;
361  for (int idim = 0; idim < arg_dim; idim++)
362  if (dgrad[idim]/cur_grad[idim] > std::numeric_limits<double>::epsilon()) {
363  valid_dgrad = true;
364  break;
365  }
366 
367  // We update B if the change in gradient is not too small
368  // relative to the value of the gradient
369  if (valid_dgrad) {
370  B_old = B;
371 
372  Vector<arg_dim> B_darg = B*darg;
373  double darg_B_darg = dot(darg, B_darg);
374 
375  double darg_dgrad = dot(darg, dgrad);
376  // Perform damping if necessary
377  if (darg_dgrad < damping_scalar*darg_B_darg) {
378  double delta = (1.0 - damping_scalar)*darg_B_darg/(darg_B_darg - darg_dgrad);
379  dgrad *= delta;
380  dgrad += (1.0 - delta)*B_darg;
381  darg_dgrad = dot(darg, dgrad);
382  }
383 
384  if (darg_dgrad <= im_tols.arg_eps*std::numeric_limits<double>::epsilon()) continue;
385 
386  // Correction to the initial Hessian guess
387  if (i == 0) {
388  B *= dgrad.norm(false)/darg_dgrad;
389  B_darg = B*darg;
390  darg_B_darg = dot(darg, B_darg);
391  }
392  B += (1.0/darg_dgrad)*(dgrad*dgrad) - (1.0/darg_B_darg)*(B_darg*B_darg);
393  }
394  }
395 
396  // Stopping conditions were NOT reached in the given number of iterations,
397  // return the last value of the argument
398  return cur_arg;
399 }

449  if (df0 >= 0.0)

451  }

458  prev_grad = cur_grad;
459

461  cur_arg, cur_dir, initial_alpha,
462  linesearch_c1, linesearch_c2,
463  im_tols, grad_rel_err_bnd,
464  cur_fval, cur_grad);

466  // Check if the change in argument was above tolerance, finish otherwise
467  if (darg.norm() < im_tols.arg_eps) return cur_arg;
468 
469  dfval += cur_fval;
470  // Check if the change in function was numerically significant, finish otherwise
471  if (std::fabs(dfval/cur_fval) <= std::numeric_limits<double>::epsilon())
472  return cur_arg;
473 
474  cur_arg += darg;
475  // Check if the value of the function is above tolerance, finish otherwise
476  if (std::fabs(cur_fval) < im_tols.fun_eps) return cur_arg;
477 
478  dgrad = cur_grad - prev_grad;
479 
480  bool valid_dgrad = false;
481  for (int idim = 0; idim < arg_dim; idim++)
482  if (dgrad[idim]/cur_grad[idim] > std::numeric_limits<double>::epsilon()) {
483  valid_dgrad = true;
484  break;
485  }
486 
487  // We update B if the change in gradient is not too small
488  // relative to the value of the gradient
489  if (valid_dgrad) {
490  B_old = B;
491 
492  Vector<arg_dim> B_darg = B*darg;
493  double darg_B_darg = dot(darg, B_darg);
494 
495  double darg_dgrad = dot(darg, dgrad);
496  double b_rec = darg_dgrad/darg_B_darg;
497 
498  Vector<arg_dim> H_dgrad;
499  Wonton::solve<arg_dim>(B, dgrad, H_dgrad);
500  double h_rec = darg_dgrad/dot(dgrad, H_dgrad);
501 
502  double l = std::min(b_rec, b_rec*h_rec);
503  double m = std::max(b_rec, b_rec*h_rec);
504 
505  double sigma2 = std::max(1.0 - 1.0/alpha, 0.5);
506  double sigma3 = exp(1);
507 
508  double phi = 1.0;
509  if (l < 1.0 - sigma2)
510  phi = sigma2/(1.0 - l);
511  else if (m > 1 + sigma3)
512  phi = sigma3/(m - 1.0);
513  else {
514  double a = 1.0/(b_rec*h_rec);
515  if ( (l >= 1.0 - sigma2) && (m <= 1.0 + sigma3) && (a > sigma3) )
516  phi = sqrt(sigma3/a);
517  }
518 
519  // Perform damping if necessary
520  if (phi != 1.0) {
521  dgrad *= phi;
522  dgrad += (1.0 - phi)*B_darg;
523  darg_dgrad = dot(darg, dgrad);
524  }
525 
526  if (darg_dgrad <= im_tols.arg_eps*std::numeric_limits<double>::epsilon()) continue;
527 
528  // Correction to the initial Hessian guess
529  if (i == 0) {
530  B *= dgrad.norm(false)/darg_dgrad;
531  B_darg = B*darg;
532  darg_B_darg = dot(darg, B_darg);
533  }
534  B += (1.0/darg_dgrad)*(dgrad*dgrad) - (1.0/darg_B_darg)*(B_darg*B_darg);
535  }
536  }
537 
538  // Stopping conditions were NOT reached in the given number of iterations,
539  // return the last value of the argument
540  return cur_arg;
