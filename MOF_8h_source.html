164  void get_plane_position(const int cellID,
165  const int matID,
166  const std::vector< MatPoly<Dim> >& mixed_polys,
167  Plane_t<Dim>& cutting_plane,
168  const bool planar_faces) const {
169  assert(Dim > 1);
170 
171  int cellMatID = std::distance(cell_mat_ids_[cellID].begin(),
172  std::find(cell_mat_ids_[cellID].begin(),
173  cell_mat_ids_[cellID].end(), matID));
174 
175  Point<Dim> cell_cen;
176  mesh_.cell_centroid(cellID, &cell_cen);
177  Vector<Dim - 1> init_guess = cartesian_to_polar(
178  cell_cen - cell_mat_centroids_[cellID][cellMatID]);
179 
180  std::function<double(const Vector<Dim - 1>&)> bfgs_obj_fun =
181  [this, &cellID, &cellMatID, &mixed_polys]
182  (const Vector<Dim - 1>& cur_arg)->double {
183  return centroid_error(cellID, cellMatID, mixed_polys,
184  polar_to_cartesian(cur_arg));
185  };
186 
187  double bfgs_obj_fun_lbnd = 0.0;
188 
189  Vector<Dim - 1> ang_min;
190  switch(mof_bfgs_alg) {
191  case BFGS: ang_min = bfgs<Dim - 1>(bfgs_obj_fun, bfgs_obj_fun_lbnd,
192  init_guess, ims_tols_[1]);
193  break;
194  case DBFGS: ang_min = dbfgs<Dim - 1>(bfgs_obj_fun, bfgs_obj_fun_lbnd,
195  init_guess, ims_tols_[1]);
196  break;
197  default: throw std::runtime_error("Unknown BFGS algorithm is selected for MOF!");
198  }
199 
200  cutting_plane.normal = polar_to_cartesian(ang_min);
201 
202  CuttingDistanceSolver<Dim, MatPoly_Clipper>
203  solve_cut_dst(mixed_polys, cutting_plane.normal, ims_tols_[0], planar_faces);
204 
205  double target_vol = cell_mat_vfracs_[cellID][cellMatID]*mesh_.cell_volume(cellID);
206  solve_cut_dst.set_target_volume(target_vol);
207  std::vector<double> clip_res = solve_cut_dst();
208  cutting_plane.dist2origin = clip_res[0];
209  }

216  std::shared_ptr<CellMatPoly<Dim>> operator()(const int cell_op_ID) const {
217  int cellID = icells_to_reconstruct[cell_op_ID];
218 
219  double dst_tol = ims_tols_[0].arg_eps;
220  // Check if the cell is single-material
221  if (cell_mat_ids_[cellID].size() == 1) {
222  std::shared_ptr< CellMatPoly<Dim> > cmp_ptr(new CellMatPoly<Dim>(cellID));
223  MatPoly<Dim> cell_matpoly;
224  cell_get_matpoly(mesh_, cellID, &cell_matpoly, dst_tol);
225  cell_matpoly.set_mat_id(cell_mat_ids_[cellID][0]);
226  cmp_ptr->add_matpoly(cell_matpoly);
227 
228  return cmp_ptr;
229  }
230 
231  // Use the nested dissections algorithm for multi-material cells.
232  // Note that nested dissections uses this instance of reconstructor
233  // to invoke get_plane_position position method. Nested dissections
234  // itself does not have its own MeshWrapper, MatPoly_Clipper, etc.,
235  // they all are reconstructor specific
236  NestedDissections<MOF, Dim, MatPoly_Splitter>
237  nested_dissections(*this, cellID, all_convex_);
238 
239  // Normally, we test all the permutations of the materials order
240  bool enable_permutations = true;
241 
242  // For two-material cells we check if the aggregated reference centroids
243  // match the cell centroid. If they do, we do NOT permute materials order
244  if (cell_mat_ids_[cellID].size() == 2) {
245  Point<Dim> cell_cen;
246  mesh_.cell_centroid(cellID, &cell_cen);
247 
248  Vector<Dim> cen_dist = cell_cen.asV();
249  for (int icmat = 0; icmat < 2; icmat++)
250  cen_dist -= cell_mat_vfracs_[cellID][icmat]*cell_mat_centroids_[cellID][icmat].asV();
251 
252  if (cen_dist.norm() <= ims_tols_[1].fun_eps)
253  enable_permutations = false;
254  }
255 
256  nested_dissections.set_cell_materials_order(enable_permutations);
257 
258  int npermutations = nested_dissections.num_materials_orders();
259  Tangram::vector<std::shared_ptr<CellMatPoly<Dim>>>
260  permutations_cellmatpoly(npermutations);
261 
262  Tangram::transform(make_counting_iterator(0),
263  make_counting_iterator(npermutations),
264  permutations_cellmatpoly.begin(), nested_dissections);
265 
266  int nmats = static_cast<int>(cell_mat_ids_[cellID].size());
267 
268  int iopt_permutation = -1;
269  double min_centroids_error = DBL_MAX;
270  for (int iperm = 0; iperm < npermutations; iperm++) {
271  std::shared_ptr<CellMatPoly<Dim>> cur_cmp_ptr =
272  permutations_cellmatpoly[iperm];
273 
274  double cur_error = 0.0;
275  for (int imat = 0; imat < nmats; imat++) {
276  int mat_id = cell_mat_ids_[cellID][imat];
277  if (!cur_cmp_ptr->is_cell_material(mat_id))
278  continue;
279 
280  const std::vector<double>& cur_moments =
281  cur_cmp_ptr->material_moments(cell_mat_ids_[cellID][imat]);
282 
283  Point<Dim> mat_centroid;
284  for (int idim = 0; idim < Dim; idim++)
285  mat_centroid[idim] = cur_moments[idim + 1]/cur_moments[0];
286 
287  // We prioritize materials with larger volumes and essentially
288  // compute a discrete L2 norm scaled by the cell volume
289  cur_error += cell_mat_vfracs_[cellID][imat]*
290  (mat_centroid - cell_mat_centroids_[cellID][imat]).norm(false);
291  }
292  cur_error = sqrt(cur_error);
293 
294  if (cur_error < min_centroids_error) {
295  iopt_permutation = iperm;
296  min_centroids_error = cur_error;
297  }
298  }
299 
300  return permutations_cellmatpoly[iopt_permutation];
301  }

336  double centroid_error(const int cellID,
337  const int cellMatID,
338  const std::vector< MatPoly<Dim> >& mixed_polys,
339  const Vector<Dim>& plane_normal) const {
340  Plane_t<Dim> cutting_plane;
341  cutting_plane.normal = plane_normal;
342 
343  double target_vol = cell_mat_vfracs_[cellID][cellMatID]*mesh_.cell_volume(cellID);
344 
345 #ifdef DEBUG
346  double vol_tol = ims_tols_[0].fun_eps;
347  //Confirm that the clipped volume is smaller than the volume of MatPoly's
348  double mixed_polys_vol = 0.0;
349  for (int ipoly = 0; ipoly < mixed_polys.size(); ipoly++)
350  mixed_polys_vol += mixed_polys[ipoly].moments()[0];
351  assert(target_vol < mixed_polys_vol + vol_tol);
352 #endif
353 
354  //Create cutting distance solver
355  CuttingDistanceSolver<Dim, MatPoly_Clipper>
356  solve_cut_dst(mixed_polys, cutting_plane.normal, ims_tols_[0], true);
357 
358  solve_cut_dst.set_target_volume(target_vol);
359  std::vector<double> clip_res = solve_cut_dst();
360 
361 #ifdef DEBUG
362  // Check if the resulting volume matches the reference value
363  double cur_vol_err = std::fabs(clip_res[1] - target_vol);
364  if (cur_vol_err > vol_tol) {
365  std::cerr << "MOF for cell " << cellID << ", testing normal ( ";
366  for (int idim = 0; idim < Dim; idim++)
367  std::cerr << plane_normal[idim] << " ";
368  std::cerr << "): given a maximum of " <<
369  ims_tols_[0].max_num_iter <<
370  " iteration(s) achieved error in volume for material " <<
371  cell_mat_ids_[cellID][cellMatID] << " is " << cur_vol_err <<
372  ", volume tolerance is " << vol_tol <<
373  ", volume of the split MatPoly's is " << mixed_polys_vol <<
374  ", target volume is " << target_vol << std::endl;
375  }
376 #endif
377 
378  Point<Dim> mat_centroid;
379  for (int idim = 0; idim < Dim; idim++)
380  mat_centroid[idim] = clip_res[idim + 2]/clip_res[1];
381 
382  return (mat_centroid - cell_mat_centroids_[cellID][cellMatID]).norm();
383  }
