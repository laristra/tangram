41 void
42 matpoly_to_r3dpoly(const MatPoly<3>& mat_poly,
43  r3d_poly& r3dpoly) {
44  //Translate coordinates of vertices to R3D format
45  const std::vector<Point3>& matpoly_vrts = mat_poly.points();
46  int nvrts = mat_poly.num_vertices();
47  r3d_rvec3* r3dized_poly_vrts;
48  r3dized_poly_vrts = new r3d_rvec3 [nvrts];
49  for (int ivrt = 0; ivrt < nvrts; ivrt++)
50  for (int ixyz = 0; ixyz < 3; ixyz++)
51  r3dized_poly_vrts[ivrt].xyz[ixyz] = matpoly_vrts[ivrt][ixyz];
52 
53  //Get faces of the polyhedron in terms of local indices of its vertices
54  int nfaces = mat_poly.num_faces();
55  r3d_int* nface_vrts;
56  nface_vrts = new r3d_int [nfaces];
57  r3d_int** r3dized_poly_faces;
58  r3dized_poly_faces = new r3d_int* [nfaces];
59  for (int iface = 0; iface < nfaces; iface++) {
60  const std::vector<int>& matpoly_faces = mat_poly.face_vertices(iface);
61  int face_nvrts = static_cast<int>(matpoly_faces.size());
62  nface_vrts[iface] = face_nvrts;
63  r3dized_poly_faces[iface] = new r3d_int [face_nvrts];
64  for (int ivrt = 0; ivrt < face_nvrts; ivrt++)
65  r3dized_poly_faces[iface][ivrt] = matpoly_faces[ivrt];
66  }
67 
68  r3d_init_poly(&r3dpoly, r3dized_poly_vrts, nvrts, r3dized_poly_faces, nface_vrts, nfaces);
69 
70  delete [] r3dized_poly_vrts;
71  delete [] nface_vrts;
72  for (int iface = 0; iface < nfaces; iface++)
73  delete [] r3dized_poly_faces[iface];
74  delete [] r3dized_poly_faces;
75 }

84 inline
85 void r3d_poly_components(const r3d_poly& r3dpoly, std::vector<r3d_poly>& poly_components,
86  const double vol_tol) {
87  const int R3D_POLY_ORDER = 1;
88  poly_components.clear();
89 
90  int total_nvrts = static_cast<int>(r3dpoly.nverts);
91  std::vector<int> vrt_marks(total_nvrts, 0);
92  std::vector< std::vector<bool> > walked_edge(total_nvrts);
93  for (int ivrt = 0; ivrt < total_nvrts; ivrt++)
94  walked_edge[ivrt].assign(3, false);
95 
96  std::vector<bool> vrt_in_cmp(total_nvrts, false);
97  std::vector<int> poly_irvt2cmp_ivrt(total_nvrts, -1);
98 
99  int icmp_first_vrt = 0;
100  r3d_real cmp_moments[R3D_NUM_MOMENTS(R3D_POLY_ORDER)];
101 
102  while(icmp_first_vrt < total_nvrts) {
103  r3d_poly cur_cmp;
104  cur_cmp.nverts = 0;
105 
106  //Walk all faces of the current components
107  int iface_first_vrt = icmp_first_vrt;
108  int iedge = 0;
109  while (iface_first_vrt < total_nvrts) {
110  int icur_vrt = iface_first_vrt;
111 
112  for (iedge = 0; iedge < 3; iedge++)
113  if (!walked_edge[iface_first_vrt][iedge])
114  break;
115 
116  //Walk the current face
117  do {
118  vrt_marks[icur_vrt]++;
119 
120  if (!vrt_in_cmp[icur_vrt]) {
121  poly_irvt2cmp_ivrt[icur_vrt] = cur_cmp.nverts;
122  cur_cmp.nverts++;
123  cur_cmp.verts[poly_irvt2cmp_ivrt[icur_vrt]] = r3dpoly.verts[icur_vrt];
124  vrt_in_cmp[icur_vrt] = true;
125  }
126 
127  int inext_vrt = r3dpoly.verts[icur_vrt].pnbrs[iedge];
128  walked_edge[icur_vrt][iedge] = true;
129 
130  int ireturn_edge;
131  for (ireturn_edge = 0; ireturn_edge < 3; ireturn_edge++)
132  if (r3dpoly.verts[inext_vrt].pnbrs[ireturn_edge] == icur_vrt)
133  break;
134 
135  iedge = (ireturn_edge + 2)%3;
136  icur_vrt = inext_vrt;
137 
138  } while (icur_vrt != iface_first_vrt);
139 
140  for (iface_first_vrt = 0; iface_first_vrt < total_nvrts; iface_first_vrt++)
141  if( (vrt_marks[iface_first_vrt] > 0) && (vrt_marks[iface_first_vrt] < 3) )
142  break;
143  }
144 
145  //Neighbors data still uses r3dpoly indices
146  for (int icmp_vrt = 0; icmp_vrt < cur_cmp.nverts; icmp_vrt++)
147  for (int iedge = 0; iedge < 3; iedge++) {
148  int old_id = cur_cmp.verts[icmp_vrt].pnbrs[iedge];
149  cur_cmp.verts[icmp_vrt].pnbrs[iedge] = poly_irvt2cmp_ivrt[old_id];
150  }
151 
152  //Confirm that the component is not empty
153  r3d_reduce(&cur_cmp, cmp_moments, R3D_POLY_ORDER);
154  if (cmp_moments[0] >= vol_tol)
155  poly_components.push_back(cur_cmp);
156 
157  //Find the first vertex of the next component
158  for (icmp_first_vrt = 0; icmp_first_vrt < total_nvrts; icmp_first_vrt++)
159  if (vrt_marks[icmp_first_vrt] == 0)
160  break;
161  }
162 }

174 void
175 r3dpoly_to_matpolys(const r3d_poly& r3dpoly,
176  std::vector< MatPoly<3> >& mat_polys,
177  const double vol_tol,
178  const double dst_tol,
179  const std::vector< Point<3> >* reference_pts = nullptr) {
180  mat_polys.clear();
181  std::vector<r3d_poly> poly_components;
182  r3d_poly_components(r3dpoly, poly_components, vol_tol);
183  if (poly_components.empty()) return;
184 
185  mat_polys.reserve(poly_components.size());
186  for (int ir3dpoly = 0; ir3dpoly < poly_components.size(); ir3dpoly++) {
187  r3d_brep* poly_brep;
188  r3d_int ncomponents;
189  r3d_poly poly_copy = poly_components[ir3dpoly];
190  r3d_init_brep(&poly_copy, &poly_brep, &ncomponents);
191 
192  if (ncomponents == 0) {
193  r3d_free_brep(&poly_brep, 0);
194  continue;
195  }
196 
197  assert(ncomponents == 1);
198 
199  int nvrts = poly_brep[0].numvertices;
200  std::vector< Point<3> > curpoly_vrts(nvrts);
201  for (int ivrt = 0; ivrt < nvrts; ivrt++)
202  for (int ixyz = 0; ixyz < 3; ixyz++)
203  curpoly_vrts[ivrt][ixyz] = poly_brep[0].vertices[ivrt].xyz[ixyz];
204 
205  int nfaces = poly_brep[0].numfaces;
206  std::vector< std::vector<int> > curpoly_faces(nfaces);
207  for (int iface = 0; iface < nfaces; iface++) {
208  int face_nverts = poly_brep[0].numvertsperface[iface];
209  curpoly_faces[iface].resize(face_nverts);
210  for (int ifv = 0; ifv < face_nverts; ifv++)
211  curpoly_faces[iface][ifv] = poly_brep[0].faceinds[iface][ifv];
212  }
213 
214  // Eliminate degeneracies that r3d has potentially introduced: vertices
215  // within dst_tol from each other are replaced by a single vertex,
216  // hanging nodes are identified and removed, degenerate faces with
217  // less than three vertices are eliminated.
218  MatPoly<3> fit_poly = natural_selection(curpoly_vrts, curpoly_faces,
219  dst_tol, reference_pts);
220  // We do not store polyhedra with less than four faces,
221  // as they are clearly degenerate
222  if (!fit_poly.points().empty()) {
223  int inew_poly = static_cast<int>(mat_polys.size());
224  mat_polys.push_back(fit_poly);
225  }
226 
227  mat_polys.shrink_to_fit();
228  r3d_free_brep(&poly_brep, ncomponents);
229  }

243 void
244 split_convex_matpoly_r3d(const MatPoly<3>& mat_poly,
245  const Plane_t<3>& cutting_plane,
246  MatPoly<3>& lower_halfspace_poly,
247  MatPoly<3>& upper_halfspace_poly,
248  std::vector<double>& lower_halfspace_moments,
249  std::vector<double>& upper_halfspace_moments,
250  double vol_tol, double dst_tol) {
251  //Translate the cutting plane to R3D format
252  r3d_plane r3d_cut_plane;
253  for (int ixyz = 0; ixyz < 3; ixyz++)
254  r3d_cut_plane.n.xyz[ixyz] = cutting_plane.normal[ixyz];
255  r3d_cut_plane.d = cutting_plane.dist2origin;
256 
257  r3d_poly r3dized_poly;
258  r3d_poly r3d_subpolys[2];
259  matpoly_to_r3dpoly(mat_poly, r3dized_poly);
260  r3d_split(&r3dized_poly, 1, r3d_cut_plane, &r3d_subpolys[1], &r3d_subpolys[0]);
261 
262  MatPoly<3>* subpoly_ptrs[2] = {&lower_halfspace_poly, &upper_halfspace_poly};
263  std::vector<double>* subpoly_moments_ptrs[2] = {&lower_halfspace_moments,
264  &upper_halfspace_moments};
265 
266  const int POLY_ORDER = 1;
267  r3d_real r3d_moments[R3D_NUM_MOMENTS(POLY_ORDER)];
268  int iempty_subpoly = -1;
269  for (int isp = 0; isp < 2; isp++) {
270  //Check if the subpoly is empty
271  //Note that if one of the subpoly's is empty, then the other is the
272  //original MatPoly, so we just need the ID (0 or 1) of the empty subpoly
273  //to get the result
274  if (r3d_subpolys[isp].nverts == 0) {
275  subpoly_ptrs[isp]->clear();
276  subpoly_moments_ptrs[isp]->assign(4, 0.0);
277  iempty_subpoly = isp;
278  break;
279  }
280 
281  //Find the moments for a subpoly
282  r3d_reduce(&r3d_subpolys[isp], r3d_moments, POLY_ORDER);
283 
284  if (r3d_moments[0] < vol_tol) {
285  subpoly_ptrs[isp]->clear();
286  subpoly_moments_ptrs[isp]->assign(4, 0.0);
287  iempty_subpoly = isp;
288  break;
289  }
290 
291  subpoly_moments_ptrs[isp]->assign(r3d_moments, r3d_moments + 4);
292  }
293 
294  //If there are no empty subpolys, we return two resulting pieces.
295  //If one of the subpolys is empty, we return the original MatPoly to
296  //ensure consistency and prevent unnecessary changes
297  if (iempty_subpoly == -1) {
298  for (int isp = 0; isp < 2; isp++) {
299  //Get a MatPoly for a subpoly
300  std::vector< MatPoly<3> > sub_matpoly;
301  r3dpoly_to_matpolys(r3d_subpolys[isp], sub_matpoly, vol_tol, dst_tol,
302  &mat_poly.points());
303  int ncomponents = static_cast<int>(sub_matpoly.size());
304  if (ncomponents > 1) {
305  // Filter out degenerate components
306  int subpoly_id = 0;
307  while (subpoly_id < sub_matpoly.size())
308  if (sub_matpoly[subpoly_id].moments()[0] >= vol_tol) subpoly_id++;
309  else sub_matpoly.erase(sub_matpoly.begin() + subpoly_id);
310  }
311 
312  if (ncomponents == 1)
313  *subpoly_ptrs[isp] = sub_matpoly[0];
314  else
315  throw std::runtime_error("Non-convex MatPoly is split using the method for convex MatPoly's!");
316  }
317  }
318  else {
319  int ifull_poly = (iempty_subpoly + 1)%2;
320  matpoly_to_r3dpoly(mat_poly, r3d_subpolys[ifull_poly]);
321  r3d_reduce(&r3d_subpolys[ifull_poly], r3d_moments, POLY_ORDER);
322  subpoly_moments_ptrs[ifull_poly]->assign(r3d_moments, r3d_moments + 4);
323 
324  *subpoly_ptrs[ifull_poly] = mat_poly;
325  }
326 }

572 void get_intersection_moments(const MatPoly<3>& mat_poly,
573  const r3d_poly& r3dpoly,
574  std::vector<double>& intersection_moments,
575  bool convex_matpoly) {
576  const int POLY_ORDER = 1;
577  r3d_real r3d_moments[R3D_NUM_MOMENTS(POLY_ORDER)];
578 
579  if (convex_matpoly) {
580  std::vector< Plane_t<3> > face_planes;
581  mat_poly.face_planes(face_planes);
582  r3d_poly intersection = r3dpoly;
583 
584  int nplanes = static_cast<int>(face_planes.size());
585  r3d_plane* r3d_face_planes = new r3d_plane [nplanes];
586  for (int iplane = 0; iplane < nplanes; iplane++) {
587  for (int ixyz = 0; ixyz < 3; ixyz++)
588  r3d_face_planes[iplane].n.xyz[ixyz] = -face_planes[iplane].normal[ixyz];
589  r3d_face_planes[iplane].d = -face_planes[iplane].dist2origin;
590  }
591 
592  r3d_clip(&intersection, r3d_face_planes, (r3d_int) nplanes);
593 
594  delete [] r3d_face_planes;
595 
596  r3d_reduce(&intersection, r3d_moments, POLY_ORDER);
597  intersection_moments.resize(4);
598  for (int im = 0; im < 4; im++)
599  intersection_moments[im] = r3d_moments[im];
600  }
601  else {
602  std::vector< MatPoly<3> > mat_poly_tets;
603  mat_poly.facetize_decompose(mat_poly_tets);
604 
605  int ntets = static_cast<int>(mat_poly_tets.size());
606  intersection_moments.assign(4, 0.0);
607  for (int itet = 0; itet < ntets; itet++) {
608  std::vector< Plane_t<3> > face_planes;
609  mat_poly_tets[itet].face_planes(face_planes);
610 
611  if(face_planes.size() < 4) continue;
612 
613  r3d_poly intersection = r3dpoly;
614  r3d_plane r3d_face_planes[4];
615  for (int iplane = 0; iplane < 4; iplane++) {
616  for (int ixyz = 0; ixyz < 3; ixyz++)
617  r3d_face_planes[iplane].n.xyz[ixyz] = -face_planes[iplane].normal[ixyz];
618  r3d_face_planes[iplane].d = -face_planes[iplane].dist2origin;
619  }
620 
621  r3d_clip(&intersection, r3d_face_planes, 4);
622 
623  r3d_reduce(&intersection, r3d_moments, POLY_ORDER);
624  for (int im = 0; im < 4; im++)
625  intersection_moments[im] += r3d_moments[im];
626  }
627  }
