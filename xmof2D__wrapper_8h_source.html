109  cell_mat_ids.begin() + offsets[icell],

155  std::shared_ptr<CellMatPoly<Dim>> operator()(const int cell_op_ID) const {
156  assert(Dim == 2);
157  assert(cell_op_ID < icells_to_reconstruct.size());
158 
159  double dst_tol = ims_tols_[0].arg_eps;
160  int cellID = icells_to_reconstruct[cell_op_ID];
161 
162  CellMatPoly<Dim>* cell_mat_poly = new CellMatPoly<Dim>(cellID);
163  const XMOF2D::BaseMesh& mesh = xmof_ir->get_base_mesh();
164  if (xmof_ir->get_cell_materials(cellID).size() > 1) {
165  xmof_ir->construct_minimesh(cellID);
166  const XMOF2D::MiniMesh& cminimesh = mesh.get_cell(cellID).get_minimesh();
167  for (int isc = 0; isc < cminimesh.ncells(); isc++) {
168  const XMOF2D::Cell& subcell = cminimesh.get_cell(isc);
169  int nvrts = subcell.nfaces();
170  std::vector<Tangram::Point<Dim>> subcell_vrts(nvrts);
171  std::vector<Tangram::Entity_kind> vrts_parentkind(nvrts);
172  std::vector<int> vrts_iparent(nvrts);
173  for (int ivrt = 0; ivrt < nvrts; ivrt++) {
174  subcell_vrts[ivrt] = Tangram::Point<Dim>(subcell.get_node_crd(ivrt).x,
175  subcell.get_node_crd(ivrt).y);
176  vrts_iparent[ivrt] = subcell.get_node(ivrt).iparent();
177  vrts_parentkind[ivrt] = (vrts_iparent[ivrt] == -1) ?
178  Tangram::Entity_kind::UNKNOWN_KIND : Tangram::Entity_kind::NODE;
179  }
180  std::vector<Tangram::Entity_kind> sides_parentkind(nvrts);
181  std::vector<int> sides_iparent(nvrts);
182  for (int iside = 0; iside < nvrts; iside++) {
183  sides_iparent[iside] = subcell.get_face(iside).iparent();
184  sides_parentkind[iside] = (sides_iparent[iside] == -1) ?
185  Tangram::Entity_kind::UNKNOWN_KIND : Tangram::Entity_kind::FACE;
186 
187  }
188  (*cell_mat_poly).add_matpoly(subcell.get_material_index(), nvrts,
189  &subcell_vrts[0], dst_tol,
190  &vrts_parentkind[0], &vrts_iparent[0],
191  &sides_parentkind[0], &sides_iparent[0]);
192  }
193  }
194  else {
195  const XMOF2D::Cell& ccell = mesh.get_cell(cellID);
196  int nvrts = ccell.nfaces();
197  std::vector<Tangram::Point<Dim>> cell_vrts(nvrts);
198  for (int ivrt = 0; ivrt < nvrts; ivrt++)
199  cell_vrts[ivrt] = Tangram::Point<Dim>(ccell.get_node_crd(ivrt).x,
200  ccell.get_node_crd(ivrt).y);
201  (*cell_mat_poly).add_matpoly(xmof_ir->get_cell_materials(cellID)[0],
202  nvrts, &cell_vrts[0], dst_tol,
203  NULL, NULL, NULL, NULL);
204  }
205 
206  return std::shared_ptr<CellMatPoly<Dim>>(cell_mat_poly);
207  }
