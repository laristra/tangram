132  assert(num_matpolys_ > 0);
133  assert(matpoly_id < num_matpolys_);
134  std::vector<int> matverts;
135  std::vector<int> const& matfaces = matpoly_faces_[matpoly_id];
136  int nf = matfaces.size();
137  for (int jf = 0; jf < nf; jf++) {
138  int f = matfaces[jf];
139  std::vector<int> const& fverts = matface_vertices_[f];
140  if (D != 2) { // order does not matter
141  for (auto const& fv : fverts)
142  if (std::find(matverts.begin(), matverts.end(), fv) == matverts.end())
143  matverts.push_back(fv);
144  } else { // order is important
145  if (matpoly_facedirs_[matpoly_id][jf] == 1) { // natural order
146  for (auto const& fv : fverts)
147  if (std::find(matverts.begin(), matverts.end(), fv) == matverts.end())
148  matverts.push_back(fv);
149  } else { // reverse the order
150  for (auto it = fverts.rbegin(); it != fverts.rend(); it++) {
151  auto const &fv = *it;
152  if (std::find(matverts.begin(), matverts.end(), fv) == matverts.end())

515 template<int D>
516 void CellMatPoly<D>::add_matpoly(int const matid,
517  double const matpoly_point0,
518  double const matpoly_point1,
519  double dst_tol,
520  Entity_kind const point0_parentkind,
521  Entity_kind const point1_parentkind,
522  int const point0_parentid,
523  int const point1_parentid) {
524 
525  assert(D == 1);
526  if (num_matpolys_ == 0)
527  dst_tol_ = dst_tol;
528  else {
529  assert(dst_tol_ == dst_tol);
530  }
531 
532  int new_matpoly_id = num_matpolys_;
533  if (std::find(cell_materialids_.begin(), cell_materialids_.end(), matid) ==
534  cell_materialids_.end())
535  cell_materialids_.push_back(matid);
536 
537  materialids_.push_back(matid);
538  matpoly_faces_.resize(num_matpolys_+1);
539  matpoly_facedirs_.resize(num_matpolys_+1);
540 
541  // Add matpoly points if they are not already there
542  // In 1D, faces are the same as vertices/points
543 
544  bool found = false;
545  int nv_ini = num_matverts_;
546  for (int i = 0; i < nv_ini; i++) {
547  Point<1>& p = matvertex_points_[i];
548  if (std::fabs(p[0] - matpoly_point0) < dst_tol_) {
549  found = true; // This point is already in CellMatPoly
550  matpoly_faces_[new_matpoly_id].push_back(i);
551  matpoly_facedirs_[new_matpoly_id].push_back(0);
552 
553  assert(matface_matpolys_[i][1] == -1); // verify slot is empty
554  matface_matpolys_[i][1] = new_matpoly_id; // new matpoly is to right of "face"
555  break;
556  }
557  }
558  if (!found) { // Introduce this point into CellMatPoly
559  matvertex_points_.push_back(Point<1>(matpoly_point0));
560  matvertex_parentkind_.push_back(point0_parentkind);
561  matvertex_parentid_.push_back(point0_parentid);
562 
563  int vert0 = num_matverts_, face0 = num_matverts_;
564  matpoly_faces_[new_matpoly_id].push_back(face0);
565  matpoly_facedirs_[new_matpoly_id].push_back(0);
566 
567  int new_matface_id = num_matfaces_;
568  matface_matpolys_.emplace_back(std::array<int, 2>({-1, -1}));
569  matface_matpolys_[new_matface_id][1] = new_matpoly_id; // new matpoly is to right of "face"
570 
571  matface_vertices_.resize(num_matfaces_+1);
572  matface_vertices_[new_matface_id].push_back(vert0);
573  matface_parentkind_.push_back(point0_parentkind);
574  matface_parentid_.push_back(point0_parentid);
575 
576  num_matverts_++;
577  num_matfaces_++;
578  }
579 
580  found = false;
581  for (int i = 0; i < nv_ini; i++) {
582  Point<1>& p = matvertex_points_[i];
583  if (std::fabs(p[0] - matpoly_point1) < dst_tol_) {
584  found = true; // This point is already in CellMatPoly
585  matpoly_faces_[new_matpoly_id].push_back(i);
586  matpoly_facedirs_[new_matpoly_id].push_back(1);
587 
588  assert(matface_matpolys_[i][0] == -1); // verify slot is empty
589  matface_matpolys_[i][0] = new_matpoly_id; // new matpoly is to left of "face"
590  break;
591  }
592  }
593  if (!found) { // Introduce this point into CellMatPoly
594  matvertex_points_.push_back(Point<1>(matpoly_point1));
595  matvertex_parentkind_.push_back(point1_parentkind);
596  matvertex_parentid_.push_back(point1_parentid);
597 
598  int vert1 = num_matverts_, face1 = num_matverts_;
599  matpoly_faces_[new_matpoly_id].push_back(face1);
600  matpoly_facedirs_[new_matpoly_id].push_back(1);
601 
602  int new_matface_id = num_matfaces_;
603 
604  matface_matpolys_.emplace_back(std::array<int, 2>({-1, -1}));
605  matface_matpolys_[new_matface_id][0] = new_matpoly_id; // new matpoly is to left of "face"
606 
607  matface_vertices_.resize(num_matfaces_+1);
608  matface_vertices_[new_matface_id].push_back(vert1);
609  matface_parentkind_.push_back(point1_parentkind);
610  matface_parentid_.push_back(point1_parentid);
611 
612  num_matverts_++;
613  num_matfaces_++;
614  }
615 
616  Point<D> vec = matpoly_point1 - matpoly_point0;

635 template<int D>
636 void CellMatPoly<D>::add_matpoly(int const matid,
637  int const numverts,
638  Point<2> const * const matpoly_points,
639  double dst_tol,
640  Entity_kind const * const points_parentkind,
641  int const * const points_parentid,
642  Entity_kind const * const faces_parentkind,
643  int const * const faces_parentid) {
644  assert(D == 2);
645  if (num_matpolys_ == 0)
646  dst_tol_ = dst_tol;
647  else {
648  assert(dst_tol_ == dst_tol);
649  }
650 
651  int new_matpoly_id = num_matpolys_;
652  if (std::find(cell_materialids_.begin(), cell_materialids_.end(), matid) ==
653  cell_materialids_.end())
654  cell_materialids_.push_back(matid);
655 
656  materialids_.push_back(matid);
657 
658  std::vector<int> matverts(numverts);
659 
660  int nv_ini = num_matverts_;
661 
662  // Add matpoly points if they are not already there
663 
664  for (int j = 0; j < numverts; j++) {
665  Point<D> const& pmat = matpoly_points[j];
666 
667  bool found = false;
668  for (int i = 0; i < nv_ini; i++) { // Note, nv remains number of original pnts
669  Point<D>& p = matvertex_points_[i];
670  // We use the provided distance tolerance to identify coincident nodes.
671  if ((p - pmat).norm() < dst_tol_) {
672  found = true; // This point is already in CellMatPoly
673  matverts[j] = i;
674  break;
675  }
676  }
677 
678  if (!found) { // Introduce this point into CellMatPoly
679  matvertex_points_.push_back(pmat);
680  Entity_kind parentkind = points_parentkind ? points_parentkind[j] :
681  Entity_kind::UNKNOWN_KIND;
682  matvertex_parentkind_.push_back(parentkind);
683  int parentid = points_parentid ? points_parentid[j] : -1;
684  matvertex_parentid_.push_back(parentid);
685  matverts[j] = matvertex_points_.size()-1;
686  num_matverts_++;
687  }
688  }
689 
690  // Add faces
691  int numfaces = numverts; // In 2D, number of "faces", which are really edges,
692  // // is the same as number of vertices
693 
694  std::vector<int> matfaces(numfaces);
695  std::vector<int> matfacedirs(numfaces);
696 
697  int nmf_ini = num_matfaces_;
698  for (int j = 0; j < numfaces; j++) {
699  std::vector<int> mfverts(2);
700  mfverts[0] = matverts[j];
701  mfverts[1] = matverts[(j+1)%numverts];
702 
703  bool found = false;
704  for (int i = 0; i < nmf_ini; i++) {
705  std::vector<int> const& mfverts2 = matface_vertices_[i];
706 
707  if (mfverts[0] == mfverts2[1] && mfverts[1] == mfverts2[0]) {
708  found = true;
709  matfaces[j] = i;
710  matfacedirs[j] = 0;
711  assert(matface_matpolys_[i][1] == -1); // verify slot is empty
712  matface_matpolys_[i][1] = new_matpoly_id;
713  break;
714  }
715  }
716 
717  if (!found) { // create a face
718  matface_vertices_.push_back(mfverts);
719  Entity_kind parentkind = faces_parentkind ? faces_parentkind[j] :
720  Entity_kind::UNKNOWN_KIND;
721  matface_parentkind_.push_back(parentkind);
722  int parentid = faces_parentid ? faces_parentid[j] : -1;
723  matface_parentid_.push_back(parentid);
724  matfaces[j] = num_matfaces_;
725  matfacedirs[j] = 1;
726 
727  matface_matpolys_.emplace_back(std::array<int, 2>({-1, -1}));
728  matface_matpolys_[num_matfaces_][0] = new_matpoly_id;
729  num_matfaces_++;
730  }
731  }
732 
733  matpoly_faces_.push_back(matfaces);
734  matpoly_facedirs_.push_back(matfacedirs);
735 
736 
737  // Geometric center of polygon; Assume that polygon is NOT so
738  // non-convex that the center is outside the polygon
739 
740  Point<D> cen;
741  for (int j = 0; j < numverts; j++)
742  cen += matpoly_points[j];
743  cen /= numverts;
744 
745  double volume = 0;
746  Point<D> centroid;
747  for (int j = 0; j < numverts; j++) {
748  Vector<D> vec0 = matpoly_points[(j+1)%numverts] - matpoly_points[j];
749  Vector<D> vec1 = cen - matpoly_points[j];
750  double trivolume = 0.5*cross(vec0, vec1);
751  volume += trivolume;
752  Point<D> tricen = (matpoly_points[j] + matpoly_points[(j+1)%numverts] +
753  cen)/3.0;
754  centroid += trivolume*tricen;
755  }
756  matpoly_volumes_.push_back(volume);
757 
758  centroid /= volume;
759  matpoly_centroids_.push_back(centroid);
760 
761  num_matpolys_++;
762 } // add_matpoly for 2D

817 
818  for (int i = 0; i < nv_ini; i++) {
819  Point<D>& p = matvertex_points_[i];
820  // We use the provided distance tolerance to identify coincident nodes.
821  if ((p - pmat).norm() < dst_tol_) {
822  found = true; // This point is already in CellMatPoly
823  matverts[j] = i;
824  break;
825  }
826  }
827 
828  if (!found) { // Introduce this point into CellMatPoly
829  matvertex_points_.push_back(pmat);
830  Entity_kind parentkind = points_parentkind ? points_parentkind[j] :
831  Entity_kind::UNKNOWN_KIND;
832  matvertex_parentkind_.push_back(parentkind);
833  int parentid = points_parentid ? points_parentid[j] : -1;
834  matvertex_parentid_.push_back(parentid);
835  matverts[j] = num_matverts_;
836  num_matverts_++;
837  }
838  }
839 
840  // Add faces
841 
842  std::vector<int> matfaces(numfaces);
843  std::vector<int> matfacedirs(numfaces);
844 
845  int nmf_ini = num_matfaces_;
846 
847  int offset = 0;
848  for (int j = 0; j < numfaces; j++) {
849  int nmfv = numfaceverts[j];
850 
851  std::vector<int> mfverts(nmfv);
852  for (int i = 0; i < nmfv; i++)
853  mfverts[i] = matverts[faces_point_ids[offset+i]];
854 
855  offset += nmfv;
856 
857  // First check if an existing face matches the points of this face.
858  // Since we stipulated that the faces of the input material
859  // polyhedron to this routine must be specified such that their
860  // normal points out of the polyhedron, a face, if it exists, must
861  // point out of a previously created polyhedron and into the one
862  // that we are trying to create. Thus we check for a matching face
863  // with vertices reversed
864 
865  bool found = false;
866  for (int i = 0; i < nmf_ini; i++) {
867 
868  // Get vertices of existing face
869 
870  std::vector<int> const& mfverts2 = matface_vertices_[i];
871  int nmfv2 = mfverts2.size();
872  if (nmfv != nmfv2) continue; // No match if number of verts don't match
873 
874  // Check if all vertices of the two faces match
875 
876  for (int k = 0; k < nmfv; k++) {
877  if (mfverts[0] == mfverts2[k]) {
878 
879  // Found one matching vertex
880  // Check if the rest of the face vertices match (in reverse
881  // since the two material polyhedra will use the face in
882  // opposite directions)
883 
884  bool allmatch = true;
885  for (int m = 1; m < nmfv; m++) {
886  if (mfverts[m] != mfverts2[(k-m+nmfv)%nmfv]) {
887  allmatch = false;
888  break;
889  }
890  }
891 
892  // Found a matching face
893  if (allmatch) {
894  found = true;
895  matfaces[j] = i;
896  matfacedirs[j] = 0;
897  assert(matface_matpolys_[i][1] == -1);
898  matface_matpolys_[i][1] = new_matpoly_id;
899  break;
900  }
901  }
902  }
903 
904  if (found)
905  break;
906  } // for (int i = 0; i < ncmf_ini; i++)
907 
908  if (!found) { // create a new matface
909  matface_vertices_.push_back(mfverts);
910  Entity_kind parentkind = faces_parentkind ? faces_parentkind[j] :
911  Entity_kind::UNKNOWN_KIND;
912  matface_parentkind_.push_back(parentkind);
913  int parentid = faces_parentid ? faces_parentid[j] : -1;
914  matface_parentid_.push_back(parentid);
915  matfaces[j] = num_matfaces_;
916  matfacedirs[j] = 1;
917 
918  matface_matpolys_.emplace_back(std::array<int, 2>({-1, -1}));
919  matface_matpolys_[num_matfaces_][0] = new_matpoly_id;
920  num_matfaces_++;
921  }
922  }
923 
924  matpoly_faces_.push_back(matfaces);
925  matpoly_facedirs_.push_back(matfacedirs);
926 
927 
928  // Compute geometric center of points. Assume that the cell is NOT so
929  // distorted that this point is outside the cell
930 
931  Point<3> cen;
932  for (int j = 0; j < numverts; j++)
933  cen += matpoly_points[j];
934  cen /= numverts;
935 
936  double volume = 0.0;
937  Point<D> centroid;
938  for (int j = 0; j < numfaces; j++) {
939  Point<D> fcen;
940  int f = matfaces[j];
941  int dir = matfacedirs[j];
942  int nmfv = matface_vertices_[f].size();
943  for (int k = 0; k < nmfv; k++) {
944  int k2 = matface_vertices_[f][k];
945  fcen += matvertex_points_[k2];
946  }
947  fcen /= nmfv;
948 
949  for (int k = 0; k < nmfv; k++) {
950  int v1 = matface_vertices_[f][k];
951  int v2 = (dir == 1) ? matface_vertices_[f][(k+1)%nmfv] :
952  matface_vertices_[f][(k-1+nmfv)%nmfv];
953 
954  Point<3> p1 = matvertex_points_[v1];
955  Point<3> p2 = matvertex_points_[v2];
956  Vector<3> vec0 = p2 - p1;
957  Vector<3> vec1 = fcen - p1;
958  Vector<3> vec2 = cen - p1;
959  Vector<3> inward_normal = -1.0*cross(vec0, vec1);
960  double tetvolume = dot(inward_normal, vec2)/6.0;
961  volume += tetvolume;
962  Point<3> tetcentroid = (p1 + p2 + fcen + cen)/4.0;
963  centroid += tetcentroid*tetvolume;
964  }
965  }
966 
967  matpoly_volumes_.push_back(volume);
968 
969  centroid /= volume;
970  matpoly_centroids_.push_back(centroid);
971 
972  num_matpolys_++;
973 } // add_matpoly for 3D
