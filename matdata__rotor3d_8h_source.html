18 double find_matching_radius(const double tooth_len,
19  const double tooth_depth,
20  const int nteeth) {
21  double l = 0.5*tooth_len, h = 0.5*tooth_depth;
22  double t = cos(PI/(2*nteeth));
23 
24  double rad = sqrt( pow2(h) +
25  ( pow2(l)*(t + 1) + l*sqrt( pow2(l)*pow2(t + 1) + 4*pow2(h)*(1 - pow2(t)) ) ) /
26  ( 1 - pow2(t) ) );
27 
28  return rad;
29 }
