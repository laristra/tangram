65  void set_volume_fractions(std::vector<int> const& cell_num_mats,
66  std::vector<int> const& cell_mat_ids,
67  std::vector<double> const& cell_mat_volfracs,
68  std::vector<Point<Dim>>

125  void get_plane_position(const int cellID,
126  const int matID,
127  const std::vector< MatPoly<Dim> >& mixed_polys,
128  Plane_t<Dim>& cutting_plane,
129  const bool planar_faces) const {
130  double vol_tol = ims_tols_[0].fun_eps;
131 
132  std::vector<int> istencil_cells;
133  mesh_.cell_get_node_adj_cells(cellID, Entity_type::ALL, &istencil_cells);
134  istencil_cells.insert(istencil_cells.begin(), cellID);
135  int nsc = static_cast<int>(istencil_cells.size());
136 
137  // Create stencil for the volume fractions gradient: the first entry corresponds
138  // to the current cell, the rest correspond to all its neighbors through the nodes
139  std::vector<double> stencil_vfracs(nsc);
140  std::vector< Point<Dim> > stencil_centroids(nsc);
141  for (int isc = 0; isc < nsc; isc++) {
142  const std::vector<int>& cur_mat_ids = cell_mat_ids_[istencil_cells[isc]];
143  if (cur_mat_ids.empty()) {
144  std::string err_msg = "VOF requires material data in the cells ";
145  err_msg += "neighboring the multi-material cells!\n";
146  err_msg += "No material data is provided for cell " + std::to_string(istencil_cells[isc]);
147  err_msg += " when reconstruction is requested for cell " + std::to_string(cellID) + "!\n";
148  throw std::runtime_error(err_msg);
149  }
150  int local_id = std::distance(cur_mat_ids.begin(),
151  std::find(cur_mat_ids.begin(), cur_mat_ids.end(), matID));
152 
153  if (local_id != cur_mat_ids.size())
154  stencil_vfracs[isc] = cell_mat_vfracs_[istencil_cells[isc]][local_id];
155 
156  mesh_.cell_centroid(istencil_cells[isc], &stencil_centroids[isc]);
157  }
158 
159  double target_vol = stencil_vfracs[0]*mesh_.cell_volume(cellID);
160  // Use least squares to compute the gradient
161  cutting_plane.normal = -ls_gradient(stencil_centroids, stencil_vfracs);
162 
163  double grad_norm = cutting_plane.normal.norm();
164  if (is_equal(grad_norm, 0.0)) {
165  // Zero gradient: we choose SLIC-like plane orientation
166  // Normal is set in the direction of the x-axis
167  cutting_plane.normal.axis(0);
168  }
169  else
170  cutting_plane.normal /= grad_norm;
171 
172  //Create cutting distance solver
173  CuttingDistanceSolver<Dim, MatPoly_Clipper>
174  solve_cut_dst(mixed_polys, cutting_plane.normal, ims_tols_[0], planar_faces);
175 
176  solve_cut_dst.set_target_volume(target_vol);
177  std::vector<double> clip_res = solve_cut_dst();
178  cutting_plane.dist2origin = clip_res[0];
179 
180 #ifdef DEBUG
181  // Check if the resulting volume matches the reference value
182  double cur_vol_err = std::fabs(clip_res[1] - target_vol);
183  if (cur_vol_err > vol_tol)
184  std::cerr << "VOF for cell " << cellID << ": given a maximum of " << ims_tols_[0].max_num_iter <<
185  " iteration(s) achieved error in volume for material " <<
186  matID << " is " << cur_vol_err << ", volume tolerance is " << vol_tol << std::endl;
187 #endif
188  }

195  std::shared_ptr<CellMatPoly<Dim>> operator()(const int cell_op_ID) const {
196  int cellID = icells_to_reconstruct[cell_op_ID];
197 
198  double dst_tol = ims_tols_[0].arg_eps;
199  // Check if the cell is single-material
200  if (cell_mat_ids_[cellID].size() == 1) {
201  std::shared_ptr< CellMatPoly<Dim> > cmp_ptr(new CellMatPoly<Dim>(cellID));
202  MatPoly<Dim> cell_matpoly;
203  cell_get_matpoly(mesh_, cellID, &cell_matpoly, dst_tol);
204  cell_matpoly.set_mat_id(cell_mat_ids_[cellID][0]);
205  cmp_ptr->add_matpoly(cell_matpoly);
206 
207  return cmp_ptr;
208  }
209 
210  // Use the nested dissections algorithm for multi-material cells.
211  // Note that nested dissections uses this instance of reconstructor
212  // to invoke get_plane_position position method. Nested dissections
213  // itself does not have its own MeshWrapper, MatPoly_Clipper, etc.,
214  // they all are reconstructor specific
215  NestedDissections<VOF, Dim, MatPoly_Splitter>
216  nested_dissections(*this, cellID, all_convex_);
217 
218  // We clip material in the same order they are given for the cell
219  // Note that this is the order of local materials, not material
220  // indices. Nested dissections uses cell_materials method to get
221  // actual material indices.
222  nested_dissections.set_cell_materials_order(false);
223 
224  return nested_dissections();
225  }
