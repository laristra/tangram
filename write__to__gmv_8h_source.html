23  int nmats,
24  std::vector<int> const& cell_num_mats,
25  std::vector<int> const& cell_mat_ids,
26  std::vector<std::shared_ptr<CellMatPoly<D>>> cellmatpoly_list,
27  std::string filename) {
28  int nc = mesh.num_entities(Entity_kind::CELL, Entity_type::PARALLEL_OWNED);
29  std::vector<int> cell_offsets(nc);
30  cell_offsets[0] = 0;
31  for (int c = 1; c < nc; c++)
32  cell_offsets[c] = cell_offsets[c-1] + cell_num_mats[c-1];
33 
34  std::ofstream fout(filename);
35  fout << std::scientific;
36  fout.precision(17);
37 
38  fout << "gmvinput ascii" << std::endl;
39  fout << "codename Tangram" << std::endl;
40  fout << "simdate 01/01/01" << std::endl;
41 
42  // Make a list of unique points in the mesh and cell mat poly
43  // structures combined
44 
45  int np = mesh.num_entities(Entity_kind::NODE, Entity_type::PARALLEL_OWNED);
46  int nmatpnts = np;
47  for (int c = 0; c < nc; c++) {
48  CellMatPoly<D> *cellmatpoly = cellmatpoly_list[c].get();
49 
50  if (cellmatpoly) { // Mixed cell
51  int ncp = cellmatpoly->num_matvertices();
52  for (int i = 0; i < ncp; i++)
53  if (cellmatpoly->matvertex_parent_kind(i) != Entity_kind::NODE)
54  nmatpnts++;
55  }
56  }
57 
58  std::vector<Point<D>> points(nmatpnts);
59 
60  for (int i = 0; i < np; i++)
61  mesh.node_get_coordinates(i, &(points[i]));
62 
63  nmatpnts = np; // reset nmatpnts so it can be used as a counter
64  for (int c = 0; c < nc; c++) {
65  CellMatPoly<D> *cellmatpoly = cellmatpoly_list[c].get();
66 
67  if (cellmatpoly) { // Mixed cell
68  int ncp = cellmatpoly->num_matvertices();
69  for (int i = 0; i < ncp; i++) {
70  if (cellmatpoly->matvertex_parent_kind(i) != Entity_kind::NODE) {
71  points[nmatpnts] = cellmatpoly->matvertex_point(i);
72  nmatpnts++;
73  }
74  }
75  }
76  }
77 
78  fout << "nodev " << nmatpnts << std::endl;
79  for (int ip = 0; ip < nmatpnts; ip++) {
80  fout << points[ip];
81  if (D == 2) // GMV requires us to output 3 coordinates
82  fout << " " << 0.000000;
83  fout << std::endl;
84  }
85 
86 
87  // Count the number of polygons we will write out
88 
89  int npoly = 0;
90  for (int c = 0; c < nc; c++) {
91  CellMatPoly<D> *cellmatpoly = cellmatpoly_list[c].get();
92  if (cellmatpoly)
93  npoly += cellmatpoly->num_matpolys();
94  else
95  npoly += 1;
96  }
97 
98  fout << "cells " << npoly << std::endl;
99 
100  for (int c = 0; c < nc; c++) {
101  CellMatPoly<D> *cellmatpoly = cellmatpoly_list[c].get();
102 
103  if (cellmatpoly) {
104 
105  // Write out the material polyhedra
106  int nmp = cellmatpoly->num_matpolys();
107  for (int i = 0; i < nmp; i++) {
108 
109  if (D == 1 || D == 2) {
110  std::vector<int> mverts = cellmatpoly->matpoly_vertices(i);
111  fout << "general 1 " << mverts.size() << " ";
112  for (auto n : mverts) {
113  if (cellmatpoly->matvertex_parent_kind(n) == Entity_kind::NODE)
114  fout << cellmatpoly->matvertex_parent_id(n)+1 << " ";
115  else {
116  Point<D> const& pnt = cellmatpoly->matvertex_point(n);
117  int np = mesh.num_entities(Entity_kind::NODE);
118  for (int j = np; j < nmatpnts; j++) {
119  if (points[j] == pnt) {
120  fout << j+1 << " ";
121  break;
122  }
123  }
124  }
125  }
126  fout << std::endl;
127  } else if (D == 3) {
128  std::vector<int> const& mfaces = cellmatpoly->matpoly_faces(i);
129  fout << "general " << mfaces.size() << std::endl;
130  for (auto f : mfaces) {
131  std::vector<int> const& mfverts = cellmatpoly->matface_vertices(f);
132  fout << mfverts.size() << " ";
133  }
134  fout << std::endl;
135  for (auto f : mfaces) {
136  std::vector<int> const& mfverts = cellmatpoly->matface_vertices(f);
137  int nfv = mfverts.size();
138  int dir, mp0, mp1;
139  cellmatpoly->matface_matpolys(f, &mp0, &mp1);
140  if (mp0 == i) { // Natural order of vertices
141  for (int j = 0; j < nfv; j++) {
142  int n = mfverts[j];
143  if (cellmatpoly->matvertex_parent_kind(n) == Entity_kind::NODE)
144  fout << cellmatpoly->matvertex_parent_id(n)+1 << " ";
145  else {
146  Point<D> const& pnt = cellmatpoly->matvertex_point(n);
147  int np = mesh.num_entities(Entity_kind::NODE);
148  for (int j = np; j < nmatpnts; j++) {
149  if (points[j] == pnt) {
150  fout << j+1 << " ";
151  break;
152  }
153  }
154  }
155  }
156  fout << std::endl;
157  } else { // Reverse order of vertices
158  for (int j = 0; j < nfv; j++) {
159  int n = mfverts[nfv-j-1];
160  if (cellmatpoly->matvertex_parent_kind(n) == Entity_kind::NODE)
161  fout << cellmatpoly->matvertex_parent_id(n)+1 << " ";
162  else {
163  Point<D> const& pnt = cellmatpoly->matvertex_point(n);
164  int np = mesh.num_entities(Entity_kind::NODE);
165  for (int j = np; j < nmatpnts; j++) {
166  if (points[j] == pnt) {
167  fout << j+1 << " ";
168  break;
169  }
170  }
171  }
172  }
173  fout << std::endl;
174  }
175  } // for (auto f : mfaces)
176  } // else if (D == 3)
177  } // for (i = 0; i < nmp; i++)
178  } else {
179  if (D == 1 || D == 2) {
180  // Write out the cell
181  std::vector<int> cverts;
182  mesh.cell_get_nodes(c, &cverts);
183  // write cell out as polygons (even if its a quad or tri)
184  fout << "general 1 " << cverts.size() << " ";
185  for (auto n : cverts)
186  fout << n+1 << " ";
187  fout << std::endl;
188  } else if (D == 3) {
189  int nf;
190  std::vector<int> cfaces;
191  std::vector<int> cfdirs;
192  mesh.cell_get_faces_and_dirs(c, &cfaces, &cfdirs);
193  fout << "general " << cfaces.size() << std::endl;
194  for (auto f : cfaces) {
195  std::vector<int> fverts;
196  mesh.face_get_nodes(f, &fverts);
197  fout << fverts.size() << " ";
198  }
199  fout << std::endl;
200  int j = 0;
201  for (auto f : cfaces) {
202  std::vector<int> fverts;
203  mesh.face_get_nodes(f, &fverts);
204  int nfverts = fverts.size();
205  if (cfdirs[j] == 1) {
206  for (int k = 0; k < nfverts; k++)
207  fout << fverts[k]+1 << " ";
208  fout << std::endl;
209  } else {
210  for (int k = 0; k < nfverts; k++)
211  fout << fverts[nfverts-k-1]+1 << " ";
212  fout << std::endl;
213  }
214  }
215  }
216  } // if (cellmatpoly) ... else
217  } // for (int c = 0; c < nc; c++)
218 
219  // Write out material ID of polygons
220 
221  fout << "material " << std::endl;
222  fout << nmats << " 0" << std::endl;
223  for (int i = 0; i < nmats; i++)
224  fout << "mat" << i+1 << std::endl;
225 
226  for (int c = 0; c < nc; c++) {
227  CellMatPoly<D> *cellmatpoly = cellmatpoly_list[c].get();
228  if (cellmatpoly) {
229  for (int i = 0; i < cellmatpoly->num_matpolys(); i++)
230  fout << cellmatpoly->matpoly_matid(i)+1 << " ";
231  } else {
232  int ofst = cell_offsets[c];
233  int matid = cell_mat_ids[ofst];
234  fout << matid+1 << " ";
235  }
236  }
237  fout << std::endl;
238 
239  fout << "endgmv" << std::endl;
240 }

242 //Simplified version: outputs CellMatPoly's list. If CellMatPoly's list
243 //was obtained from a reconstructor, it corresponds to writing out only
244 //multi-material cells. Does not require the base mesh and is therefore
245 //convenient for debugging and/or when a single CellMatPoly needs to be
246 //written out.
247 //Note that unlike in the full version, nodes and faces of CellMatPoly's
248 //are written as is and will be duplicated if shared by different CellMatPoly's.
249 template<int D>
250 void write_to_gmv(const std::vector<std::shared_ptr<CellMatPoly<D>>>& cellmatpoly_list,
251  const std::string& filename) {
252  std::ofstream fout(filename);
253  fout << std::scientific;
254  fout.precision(17);
255 
256  fout << "gmvinput ascii" << std::endl;
257  fout << "codename Tangram" << std::endl;
258  fout << "simdate 01/01/01" << std::endl;
259 
260  int npoly = 0;
261  int nmats = 0;
262  std::vector<Point<D>> points;
263  for (int imp = 0; imp < cellmatpoly_list.size(); imp++) {
264  CellMatPoly<D> *cellmatpoly = cellmatpoly_list[imp].get();
265 
266  if (cellmatpoly) { // Mixed cell
267  int ncp = cellmatpoly->num_matvertices();
268  for (int i = 0; i < ncp; i++)
269  points.push_back(cellmatpoly->matvertex_point(i));
270  npoly += cellmatpoly->num_matpolys();
271  const std::vector<int>& matpoly_matids = cellmatpoly->matpoly_matids();
272  int max_matid = *std::max_element(matpoly_matids.begin(), matpoly_matids.end());
273  if (max_matid >= nmats) nmats = max_matid + 1;
274  }
275  }
276 
277  int nmatpnts = static_cast<int>(points.size());
278  fout << "nodev " << nmatpnts << std::endl;
279  for (int ip = 0; ip < nmatpnts; ip++) {
280  fout << points[ip];
281  if (D == 2) // GMV requires us to output 3 coordinates
282  fout << " " << 0.000000;
283  fout << std::endl;
284  }
285 
286  fout << "cells " << npoly << std::endl;
287  int pts_offset = 0;
288  for (int imp = 0; imp < cellmatpoly_list.size(); imp++) {
289  CellMatPoly<D> *cellmatpoly = cellmatpoly_list[imp].get();
290  if (!cellmatpoly)
291  continue;
292 
293  // Write out the material polyhedra
294  int nmp = cellmatpoly->num_matpolys();
295  for (int i = 0; i < nmp; i++) {
296  if (D == 1 || D == 2) {
297  std::vector<int> mverts = cellmatpoly->matpoly_vertices(i);
298  fout << "general 1 " << mverts.size() << " ";
299  for (auto n : mverts)
300  fout << pts_offset + n + 1 << " ";
301  fout << std::endl;
302  }
303  else if (D == 3) {
304  std::vector<int> const& mfaces = cellmatpoly->matpoly_faces(i);
305  fout << "general " << mfaces.size() << std::endl;
306  for (auto f : mfaces) {
307  std::vector<int> const& mfverts = cellmatpoly->matface_vertices(f);
308  fout << mfverts.size() << " ";
309  }
310  fout << std::endl;
311  for (auto f : mfaces) {
312  std::vector<int> const& mfverts = cellmatpoly->matface_vertices(f);
313  for (auto n : mfverts)
314  fout << pts_offset + n + 1 << " ";
315  fout << std::endl;
316  }
317  } // else if (D == 3)
318  } // for (i = 0; i < nmp; i++)
319  pts_offset += cellmatpoly->num_matvertices();
320  } // for (int imp = 0; imp < cellmatpoly_list.size(); imp++)
321 
322  // Write out material ID of polygons
323 
324  fout << "material " << std::endl;
325  fout << nmats << " 0" << std::endl;
326  for (int i = 0; i < nmats; i++)
327  fout << "mat" << i + 1 << std::endl;
328 
329  for (int imp = 0; imp < cellmatpoly_list.size(); imp++) {
330  CellMatPoly<D> *cellmatpoly = cellmatpoly_list[imp].get();
331  if (cellmatpoly) {
332  for (int i = 0; i < cellmatpoly->num_matpolys(); i++)
333  fout << cellmatpoly->matpoly_matid(i) + 1 << " ";
334  }
335  }
336  fout << std::endl;
337 
338  fout << "endgmv" << std::endl;
339 }
