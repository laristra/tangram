36 void advance_moments(const Mesh_Wrapper& mesh,
37  const std::vector<int>& mesh_mat_IDs,
38  const std::vector<double>& mesh_mat_vols,
39  const std::function<Tangram::Vector<D>(double,
40  const Tangram::Point<D>&)>& velocity,
41  double t_prev,
42  double dt,
43  const std::vector<int>& cell_num_mats,
44  const std::vector<int>& cell_mat_ids,
45  std::vector<double>& cell_mat_volfracs,
46  std::vector< Tangram::Point<D> >& cell_mat_centroids,
47  double vol_tol) {
48 
49  int ncells = mesh.num_owned_cells() + mesh.num_ghost_cells();
50  std::vector<int> offsets(ncells, 0);
51  for (int icell = 0; icell < ncells - 1; icell++)
52  offsets[icell + 1] = offsets[icell] + cell_num_mats[icell];
53 
54  // Advance the positions of centroids
55  for (int icell = 0; icell < ncells; icell++)
56  for (int icmat = 0; icmat < cell_num_mats[icell]; icmat++)
57  cell_mat_centroids[offsets[icell] + icmat] = Tangram::runge_kutta_4(
58  velocity, cell_mat_centroids[offsets[icell] + icmat], t_prev, dt);
59 
60  // Calculate total volume fraction of material each cell was assigned
61  std::vector<double> total_mat_volfracs(ncells, 0.0);
62  for (int icell = 0; icell < ncells; icell++)
63  for (int icmat = 0; icmat < cell_num_mats[icell]; icmat++)
64  total_mat_volfracs[icell] += cell_mat_volfracs[offsets[icell] + icmat];
65 
66  // Determine cells that have too much material and push it out
67  for (int icell = 0; icell < ncells; icell++) {
68  double cell_size = mesh.cell_volume(icell);
69  if (total_mat_volfracs[icell] < 1.0 + vol_tol/cell_size)
70  continue;
71 
72  // Sort material indices in the order of increasing volume fractions
73  int ncmats = cell_num_mats[icell];
74  std::vector<int> sorted_icmats = Tangram::sorted_indices(std::vector<double>(
75  cell_mat_volfracs.begin() + offsets[icell],
76  cell_mat_volfracs.begin() + offsets[icell] + ncmats), true);
77 
78  std::vector<int> iadj_cells;
79  mesh.cell_get_node_adj_cells(icell, Tangram::Entity_type::PARALLEL_OWNED, &iadj_cells);
80  for (int iscm = 0; iscm < ncmats; iscm++) {
81  int icmat = sorted_icmats[iscm];
82  int mat_id = cell_mat_ids[offsets[icell] + icmat];
83  double extra_mat_vol = cell_size*cell_mat_volfracs[offsets[icell] + icmat]*
84  (1.0 - 1.0/total_mat_volfracs[icell]);
85 
86  std::vector<int> ireceivers = iadj_cells;
87  while (extra_mat_vol >= vol_tol/ncmats) {
88  std::vector<int> ireceivers_upd;
89  double slice_vol = extra_mat_vol/ireceivers.size();
90  for (int irc = 0; irc < ireceivers.size(); irc++) {
91  int iadj_cell = ireceivers[irc];
92  double adj_cell_size = mesh.cell_volume(iadj_cell);
93  double adjc_capacity = adj_cell_size*(1.0 - total_mat_volfracs[iadj_cell]);
94  if (adjc_capacity < vol_tol)
95  continue;
96 
97  int adjc_matid = std::distance(cell_mat_ids.begin() + offsets[iadj_cell],
98  std::find(cell_mat_ids.begin() + offsets[iadj_cell],
99  cell_mat_ids.begin() + offsets[iadj_cell] + cell_num_mats[iadj_cell],
100  mat_id));
101  if (adjc_matid == cell_num_mats[iadj_cell])
102  continue;
103 
104  double vol_delta = (adjc_capacity < slice_vol) ? adjc_capacity : slice_vol;
105  adjc_capacity -= vol_delta;
106 
107  double adjc_vf_delta = vol_delta/adj_cell_size;
108  cell_mat_volfracs[offsets[iadj_cell] + adjc_matid] += adjc_vf_delta;
109  total_mat_volfracs[iadj_cell] += adjc_vf_delta;
110  if (adjc_capacity >= vol_tol)
111  ireceivers_upd.push_back(iadj_cell);
112 
113  extra_mat_vol -= vol_delta;
114  double vf_delta = vol_delta/cell_size;
115  cell_mat_volfracs[offsets[icell] + icmat] -= vf_delta;
116  total_mat_volfracs[icell] -= vf_delta;
117  }
118 
119  if (!ireceivers_upd.empty())
120  ireceivers = ireceivers_upd;
121  else {
122  double extra_vf_delta = extra_mat_vol/cell_size;
123  cell_mat_volfracs[offsets[icell] + icmat] -= extra_vf_delta;
124  total_mat_volfracs[icell] -= extra_vf_delta;
125  break;
126  }
127  }
128  }
129  }
130 
131  // Determine how much material is left to be distributed
132  std::vector<double> extra_mesh_mat_vols = mesh_mat_vols;
133  for (int icell = 0; icell < ncells; icell++) {
134  double cell_size = mesh.cell_volume(icell);
135  for (int icmat = 0; icmat < cell_num_mats[icell]; icmat++) {
136  int imatid = std::distance(mesh_mat_IDs.begin(),
137  std::find(mesh_mat_IDs.begin(), mesh_mat_IDs.end(),
138  cell_mat_ids[offsets[icell] + icmat]));
139  extra_mesh_mat_vols[imatid] -= cell_mat_volfracs[offsets[icell] + icmat]*cell_size;
140  }
141  }
142 
143  // Determine cells that have too little material and fill them up
144  for (int icell = 0; icell < ncells; icell++) {
145  double cell_size = mesh.cell_volume(icell);
146  if (total_mat_volfracs[icell] > 1.0 - vol_tol/cell_size)
147  continue;
148 
149  int ncmats = cell_num_mats[icell];
150 
151  // Find global leftover volumes for contained materials
152  std::vector<int> imatids(ncmats);
153  double total_extra_mat_vol = 0.0;
154  for (int icmat = 0; icmat < ncmats; icmat++) {
155  imatids[icmat] = std::distance(mesh_mat_IDs.begin(),
156  std::find(mesh_mat_IDs.begin(), mesh_mat_IDs.end(),
157  cell_mat_ids[offsets[icell] + icmat]));
158  if (extra_mesh_mat_vols[imatids[icmat]] > 0.0)
159  total_extra_mat_vol += extra_mesh_mat_vols[imatids[icmat]];
160  }
161 
162  // Weight the material based on the leftover volumes
163  std::vector<double> cmat_weights(ncmats);
164  if (total_extra_mat_vol < vol_tol)
165  cmat_weights.assign(ncmats, 1.0/ncmats);
166  else {
167  for (int icmat = 0; icmat < ncmats; icmat++)
168  cmat_weights[icmat] = extra_mesh_mat_vols[imatids[icmat]]/total_extra_mat_vol;
169  }
170 
171  //Fill the cell up to full size
172  double empty_vf = 1.0 - total_mat_volfracs[icell];
173  for (int icmat = 0; icmat < ncmats; icmat++) {
174  if (cmat_weights[icmat] == 0.0)
175  continue;
176 
177  double vf_delta = empty_vf*cmat_weights[icmat];
178  cell_mat_volfracs[offsets[icell] + icmat] += vf_delta;
179  total_mat_volfracs[icell] += vf_delta;
180  extra_mesh_mat_vols[imatids[icmat]] -= vf_delta*cell_size;
181  }
182  }
183 
184  for (int icell = 0; icell < ncells; icell++) {
185  double mat_volfrac = 0.0;
186  for (int icmat = 0; icmat < cell_num_mats[icell]; icmat++)
187  mat_volfrac += cell_mat_volfracs[offsets[icell] + icmat];
188 
189  assert(std::fabs(mat_volfrac - 1.0) < vol_tol/mesh.cell_volume(icell));
190  }
191 }
