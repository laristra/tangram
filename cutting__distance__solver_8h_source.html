42 class CuttingDistanceSolver {
43 public:
44  CuttingDistanceSolver(const std::vector< MatPoly<D> >& matpolys,
45  const Vector<D>& plane_normal,

56  assert(target_vol >= 0.0);
57 #endif

67  std::vector<double> operator() () const {
68  double vol_tol = tolerances_.fun_eps;
69  double dst_tol = tolerances_.arg_eps;
70 
71  double dst_bnd[2] = { -DBL_MAX, DBL_MAX };
72  // Find the cutting distance corresponding to the planes passing through
73  // the nearest and the farthest vertices
74  for (int ipoly = 0; ipoly < matpolys_.size(); ipoly++) {
75  const std::vector< Point<D> >& poly_pts = matpolys_[ipoly].points();
76  for (int ivrt = 0; ivrt < matpolys_[ipoly].num_vertices(); ivrt++) {
77  // dist2origin for Plane_t is defined as dot(PO, normal), where PO
78  // is the vector from the point to the origin and is the negative of P.asV()
79  double dst_vrt = -dot(plane_normal_, poly_pts[ivrt].asV());
80  // dst_bnd is associated with the volume below the plane with dst_bnd[0]
81  // corresponding to the zero volume, so the sign is reversed
82  if (dst_vrt > dst_bnd[0]) dst_bnd[0] = dst_vrt;
83  if (dst_vrt < dst_bnd[1]) dst_bnd[1] = dst_vrt;
84  }
85  }
86 
87  if (target_vol_ < vol_tol) {
88  // Size of the clipped polys is zero: we return the distance corresponding
89  // to the nearest vertex and zero moments
90  std::vector<double> empty(D + 2, 0.0);
91  empty[0] = dst_bnd[0];
92  return empty;
93  }
94 
95  MatPoly_Clipper clip_matpolys(vol_tol);
96  clip_matpolys.set_matpolys(matpolys_, planar_faces_);
97  // Get aggregate moments of the cut MatPoly's
98  std::vector<double> full_moments = clip_matpolys.aggregated_moments();
99 
100  if (target_vol_ > full_moments[0] - vol_tol) {
101  // All the poly's are below the plane: we return the distance corresponding
102  // to the farthest vertex and aggregate moments
103  full_moments.insert(full_moments.begin(), dst_bnd[1]);
104  return full_moments;
105  }
106 
107  double d2orgn = 0.5*(dst_bnd[0] + dst_bnd[1]);
108  // As distance changes from dst_bnd[0] to dst_bnd[1], the volume changes
109  // from 0 to the aggregated volume of matpolys
110  double vol_bnd[2] = { 0.0, full_moments[0] };
111  double vol_err = full_moments[0]; // Current volume mismatch
112  double dvol = full_moments[0]; // Change in volume over secant method's iteration
113  double cur_dst_bnd[2] = { dst_bnd[0], dst_bnd[1] };
114  std::vector<double> cur_moments;
115 
116  bool use_secant = true;
117  int iter_count = 0;
118 
119  Plane_t<D> cutting_plane = {.normal = plane_normal_, .dist2origin = d2orgn};
120 
121  // We iterative solve for the cutting distance that corresponds to the target volume
122  // below the cutting plane. By default, the secant method is used.
123  // If the secant method fails, we discard its results, fall back to the initial guess
124  // and bounds, then use the bisection algorithm instead.
125  while (vol_err > vol_tol) {
126  double sec_coef;
127  if (use_secant) {
128  // Both the change in volume on the previous step and the max possible change
129  // in volume on this step should be above volume tolerance to keep using
130  // the secant method
131  use_secant = (std::fabs(dvol) > vol_tol) &&
132  (std::fabs(vol_bnd[1] - vol_bnd[0]) > vol_tol);
133  if (use_secant) {
134  sec_coef = (cur_dst_bnd[1] - cur_dst_bnd[0])/(vol_bnd[1] - vol_bnd[0]);
135  d2orgn = cur_dst_bnd[1] + sec_coef*(target_vol_ - vol_bnd[1]);
136  use_secant = ((d2orgn > dst_bnd[0]) == (d2orgn < dst_bnd[1]));
137  }
138 
139  if(!use_secant) {
140  // Secant method failed: fall back to the bisection algorithm
141  bool valid_lower_bnd = (vol_bnd[0] < target_vol_);
142  if ( valid_lower_bnd != (vol_bnd[1] > target_vol_) ) {
143  // Target volume fraction is no longer within the current bounds:
144  // reset current bounds
145  cur_dst_bnd[0] = dst_bnd[0]; cur_dst_bnd[1] = dst_bnd[1];
146  }
147  else if (!valid_lower_bnd) {
148  // vfrac_bnd[0] is the upper bound: swap the current bounds
149  std::reverse(std::begin(cur_dst_bnd), std::end(cur_dst_bnd));
150  }
151 
152  d2orgn = 0.5*(cur_dst_bnd[0] + cur_dst_bnd[1]);
153  }
154  }
155  else
156  d2orgn = 0.5*(cur_dst_bnd[0] + cur_dst_bnd[1]);
157 
158  if (std::fabs(cur_dst_bnd[1] - cur_dst_bnd[0]) < dst_tol) {
159  // The currently used method is valid and possible change in
160  // cutting distance is negligible, we can terminate
161  cur_moments.insert(cur_moments.begin(), cutting_plane.dist2origin);
162  return cur_moments;
163  }
164 
165  cutting_plane.dist2origin = d2orgn;
166  clip_matpolys.set_plane(cutting_plane);
167  cur_moments = clip_matpolys();
168 
169  vol_err = std::fabs(cur_moments[0] - target_vol_);
170  iter_count++;
171  if (iter_count > tolerances_.max_num_iter) {
172  // Max number of iterations reached: we return the current distance.
173  // Because the moments are also returned, the caller can calculate the
174  // discrepancy in volumes and take appropriate action
175  break;
176  }
177 
178  if (!use_secant) {
179  if (cur_moments[0] > target_vol_)
180  cur_dst_bnd[1] = d2orgn;
181  else
182  cur_dst_bnd[0] = d2orgn;
183  }
184  else {
185  // Compute the change in volume over this step
186  dvol = cur_moments[0] - vol_bnd[1];
187 
188  cur_dst_bnd[0] = cur_dst_bnd[1];
189  cur_dst_bnd[1] = d2orgn;
190  vol_bnd[0] = vol_bnd[1];
191  vol_bnd[1] = cur_moments[0];
192  }
193  }
194  cur_moments.insert(cur_moments.begin(), d2orgn);
195  return cur_moments;
196  }
