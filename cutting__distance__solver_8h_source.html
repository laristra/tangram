42 class CuttingDistanceSolver {
43 public:
44  CuttingDistanceSolver(const std::vector< MatPoly<D> >& matpolys,
45  const Vector<D>& plane_normal,
46  const IterativeMethodTolerances_t& tolerances,
47  const bool planar_faces = true) :
48  matpolys_(matpolys), plane_normal_(plane_normal),
49  tolerances_(tolerances), planar_faces_(planar_faces) {}
54  void set_target_volume(double const target_vol) {
55 #ifdef DEBUG
56  assert(target_vol >= 0.0);
57 #endif
58  target_vol_ = target_vol;
59  }

67  std::vector<double> operator() () const {
68  double vol_eps = tolerances_.fun_eps;
69 
70  double dst_bnd[2] = { -DBL_MAX, DBL_MAX };
71  // Find the cutting distance corresponding to the planes passing through
72  // the nearest and the farthest vertices
73  for (int ipoly = 0; ipoly < matpolys_.size(); ipoly++) {
74  const std::vector< Point<D> >& poly_pts = matpolys_[ipoly].points();
75  for (int ivrt = 0; ivrt < matpolys_[ipoly].num_vertices(); ivrt++) {
76  // dist2origin for Plane_t is defined as dot(PO, normal), where PO
77  // is the vector from the point to the origin and is the negative of P.asV()
78  double dst_vrt = -dot(plane_normal_, poly_pts[ivrt].asV());
79  // dst_bnd is associated with the volume below the plane with dst_bnd[0]
80  // corresponding to the zero volume, so the sign is reversed
81  if (dst_vrt > dst_bnd[0]) dst_bnd[0] = dst_vrt;
82  if (dst_vrt < dst_bnd[1]) dst_bnd[1] = dst_vrt;
83  }
84  }
85 
86  if (target_vol_ < vol_eps) {
87  // Size of the clipped polys is zero: we return the distance corresponding
88  // to the nearest vertex and zero moments
89  std::vector<double> empty(D + 2, 0.0);
90  empty[0] = dst_bnd[0];
91  return empty;
92  }
93 
94  MatPoly_Clipper clip_matpolys;
95  clip_matpolys.set_matpolys(matpolys_, planar_faces_);
96  // Get aggregate moments of the cut MatPoly's
97  std::vector<double> full_moments = clip_matpolys.aggregated_moments();
98 
99  if (target_vol_ > full_moments[0] - vol_eps) {
100  // All the poly's are below the plane: we return the distance corresponding
101  // to the farthest vertex and aggregate moments
102  full_moments.insert(full_moments.begin(), dst_bnd[1]);
103  return full_moments;
104  }
105 
106  double d2orgn = 0.5*(dst_bnd[0] + dst_bnd[1]);
107  // As distance changes from dst_bnd[0] to dst_bnd[1], the volume changes
108  // from 0 to the aggregated volume of matpolys
109  double vol_bnd[2] = { 0.0, full_moments[0] };
110  double vol_err = full_moments[0]; // Current volume mismatch
111  double dvol = full_moments[0]; // Change in volume over secant method's iteration
112  double cur_dst_bnd[2] = { dst_bnd[0], dst_bnd[1] };
113  std::vector<double> cur_moments;
114 
115  bool use_secant = true;
116  int iter_count = 0;
117 
118  Plane_t<D> cutting_plane = {.normal = plane_normal_, .dist2origin = d2orgn};
119 
120  // We iterative solve for the cutting distance that corresponds to the target volume
121  // below the cutting plane. By default, the secant method is used.
122  // If the secant method fails, we discard its results, fall back to the initial guess
123  // and bounds, then use the bisection algorithm instead.
124  while (vol_err > vol_eps) {
125  double sec_coef;
126  if (use_secant) {
127  // Both the change in volume on the previous step and the max possible change
128  // in volume on this step should be above volume tolerance to keep using
129  // the secant method
130  use_secant = (std::fabs(dvol) > vol_eps) &&
131  (std::fabs(vol_bnd[1] - vol_bnd[0]) > vol_eps);
132  if (use_secant) {
133  sec_coef = (cur_dst_bnd[1] - cur_dst_bnd[0])/(vol_bnd[1] - vol_bnd[0]);
134  d2orgn = cur_dst_bnd[1] + sec_coef*(target_vol_ - vol_bnd[1]);
135  use_secant = ((d2orgn > dst_bnd[0]) == (d2orgn < dst_bnd[1]));
136  }
137 
138  if(!use_secant) {
139  // Secant method failed: fall back to the bisection algorithm
140  bool valid_lower_bnd = (vol_bnd[0] < target_vol_);
141  if ( valid_lower_bnd != (vol_bnd[1] > target_vol_) ) {
142  // Target volume fraction is no longer within the current bounds:
143  // reset current bounds
144  cur_dst_bnd[0] = dst_bnd[0]; cur_dst_bnd[1] = dst_bnd[1];
145  }
146  else if (!valid_lower_bnd) {
147  // vfrac_bnd[0] is the upper bound: swap the current bounds
148  std::reverse(std::begin(cur_dst_bnd), std::end(cur_dst_bnd));
149  }
150 
151  d2orgn = 0.5*(cur_dst_bnd[0] + cur_dst_bnd[1]);
152  }
153  }
154  else
155  d2orgn = 0.5*(cur_dst_bnd[0] + cur_dst_bnd[1]);
156 
157  if (std::fabs(cur_dst_bnd[1] - cur_dst_bnd[0]) <= tolerances_.arg_eps) {
158  // The currently used method is valid and possible change in
159  // cutting distance is negligible, we can terminate
160  cur_moments.insert(cur_moments.begin(), cutting_plane.dist2origin);
161  return cur_moments;
162  }
163 
164  cutting_plane.dist2origin = d2orgn;
165  clip_matpolys.set_plane(cutting_plane);
166  cur_moments = clip_matpolys();
167 
168  vol_err = std::fabs(cur_moments[0] - target_vol_);
169  iter_count++;
170  if (iter_count > tolerances_.max_num_iter) {
171  // Max number of iterations reached: we return the current distance.
172  // Because the moments are also returned, the caller can calculate the
173  // discrepancy in volumes and take appropriate action
174  break;
175  }
176 
177  if (!use_secant) {
178  if (cur_moments[0] > target_vol_)
179  cur_dst_bnd[1] = d2orgn;
180  else
181  cur_dst_bnd[0] = d2orgn;
182  }
183  else {
184  // Compute the change in volume over this step
185  dvol = cur_moments[0] - vol_bnd[1];
186 
187  cur_dst_bnd[0] = cur_dst_bnd[1];
188  cur_dst_bnd[1] = d2orgn;
189  vol_bnd[0] = vol_bnd[1];
190  vol_bnd[1] = cur_moments[0];
191  }
192  }
193 
194  cur_moments.insert(cur_moments.begin(), d2orgn);
195  return cur_moments;
196  }
