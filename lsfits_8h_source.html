40 template<long D>
41 Vector<D> ls_gradient(std::vector<Point<D>> const & coords,
42  std::vector<double> const & vals) {
43 
44  Point<D> coord0 = coords[0];
45 
46  double val0 = vals[0];
47 
48  // There are nvals but the first is the reference point where we
49  // are trying to compute the gradient; so the matrix sizes etc
50  // will only be nvals-1
51 
52  int nvals = vals.size();
53 
54  // Each row of A contains the components of the vector from
55  // coord0 to the candidate point being used in the Least Squares
56  // approximation (X_i-X_0).
57 
58  Matrix A(nvals-1, D);
59  for (int i = 0; i < nvals-1; ++i) {
60  for (int j = 0; j < D; ++j)
61  A[i][j] = coords[i+1][j]-coord0[j];
62  }
63 
64 
65  // A is a matrix of size nvals-1 by D (where D is the space
66  // dimension). So transpose(A)*A is D by D
67 
68  Matrix AT = A.transpose();
69 
70  Matrix ATA = AT*A;
71 
72  // Each entry/row of F contains the difference between the
73  // function value at the candidate point and the function value
74  // at the point where we are computing (f-f_0)
75 
76  std::vector<double> F(nvals-1);
77  for (int i = 0; i < nvals-1; ++i)
78  F[i] = vals[i+1]-val0;
79 
80  // F is a vector of nvals. So transpose(A)*F is vector of D
81  // (where D is the space dimension)
82 
83  Vector<D> ATF = Vector<D>(AT*F);
84 
85  // Inverse of ATA
86 
87  Matrix ATAinv = ATA.inverse();
88 
89  // Gradient of length D
90 
91  return ATAinv*ATF;
92 }
