96  std::shared_ptr< CellMatPoly<Dim> > operator()(const int permutation_ID = 0) const {
97  assert(permutation_ID < cell_materials_order_.size());
98 
99  // Get material indices for the cell from the reconstructor
100  const std::vector<int>& mat_ids = reconstructor_.cell_materials(cell_id_);
101  int nmats = static_cast<int>(mat_ids.size());
102 
103  std::shared_ptr< CellMatPoly<Dim> > cmp_ptr =
104  std::make_shared< CellMatPoly<Dim> >(cell_id_);
105 
106  //MatPoly corresponding to a non-convex cell can have non-planar faces
107  MatPoly<Dim> cell_matpoly;
108  if (convex_cell_)
109  cell_matpoly = reconstructor_.cell_matpoly(cell_id_);
110  else
111  reconstructor_.cell_matpoly(cell_id_).faceted_matpoly(&cell_matpoly);
112 
113  double vol_tol = reconstructor_.iterative_methods_tolerances()[0].fun_eps;
114 
115  Plane_t<Dim> cutting_plane;
116  HalfSpaceSets_t<Dim> hs_sets; // Structure containing vectors of MatPoly's and their
117  // aggregated moments for the respective half-spaces
118 
119  //Original vector of mixed MatPoly's consists of the cell's MatPoly
120  hs_sets.upper_halfspace_set.matpolys = {cell_matpoly};
121 
122  //Create Splitter instance: assumes split MatPoly's are convex
123  MatPoly_Splitter split_matpolys(hs_sets.upper_halfspace_set.matpolys,
124  cutting_plane, true);
125 
126  for (int imat = 0; imat < nmats; imat++) {
127  int matid = mat_ids[cell_materials_order_[permutation_ID][imat]];
128 
129  MatPolySet_t<Dim>* single_mat_set_ptr;
130  //On the last iteration the remaining part is single-material,
131  //so we don't need to split it
132  if (imat == nmats - 1)
133  single_mat_set_ptr = &hs_sets.upper_halfspace_set;
134  else {
135  // Find cutting plane position: assumes all faces are planar
136  // Note that this method generally does NOT require MatPoly_Splitter,
137  // but uses MatPoly_Clipper instead. Nested dissections itself
138  // has neither MatPoly_Clipper, nor MeshWrapper, those are
139  // handled by the reconstructor that created this nested dissections
140  // instance.
141  reconstructor_.get_plane_position(cell_id_, matid,
142  hs_sets.upper_halfspace_set.matpolys,
143  cutting_plane, true);
144 
145  // On the first cut we have to decompose a non-convex MatPoly
146  // This is done inside the loop to make the first step cheaper:
147  // finding the position of the cutting plane normally requires
148  // only computation of moments, which we assume to be possible
149  // without decomposing into tetrahedrons (e.g. if r3d is used).
150  if (!convex_cell_ && (imat == 0)) {
151  hs_sets.upper_halfspace_set.matpolys.clear();
152  cell_matpoly.decompose(hs_sets.upper_halfspace_set.matpolys);
153  }
154 
155  hs_sets = split_matpolys();
156  //Single-material MatPoly's are below the plane
157  single_mat_set_ptr = &hs_sets.lower_halfspace_set;
158 
159  // Filter out MatPoly's with volumes below tolerance
160  int ismp = 0;
161  while (ismp < single_mat_set_ptr->matpolys.size())
162  if (single_mat_set_ptr->matpolys[ismp].moments()[0] > vol_tol)
163  ismp++;
164  else {
165  if (single_mat_set_ptr->matpolys[ismp].moments()[0] >
166  std::numeric_limits<double>::epsilon())
167  for (int im = 0; im < Dim + 1; im++)
168  single_mat_set_ptr->moments[im] -=
169  single_mat_set_ptr->matpolys[ismp].moments()[im];
170 
171  single_mat_set_ptr->matpolys.erase(
172  single_mat_set_ptr->matpolys.begin() + ismp);
173  }
174  }
175 
176  // Add single-material poly's below the plane to CellMatPoly
177  for (int ismp = 0; ismp < single_mat_set_ptr->matpolys.size(); ismp++) {
178  MatPoly<Dim>& cur_matpoly = single_mat_set_ptr->matpolys[ismp];
179  cur_matpoly.set_mat_id(matid);
180  cmp_ptr->add_matpoly(cur_matpoly);
181  }
182 
183  if (single_mat_set_ptr->matpolys.size() > 0)
184  cmp_ptr->assign_material_moments(matid, single_mat_set_ptr->moments);
185  }
186 
187  return cmp_ptr;
188  }
