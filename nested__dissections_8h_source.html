96  std::shared_ptr< CellMatPoly<Dim> > operator()(const int permutation_ID = 0) const {
97  assert(permutation_ID < cell_materials_order_.size());
98 
99  // Get material indices for the cell from the reconstructor
100  const std::vector<int>& mat_ids = reconstructor_.cell_materials(cell_id_);
101  int nmats = static_cast<int>(mat_ids.size());
102 
103  std::shared_ptr< CellMatPoly<Dim> > cmp_ptr =
104  std::make_shared< CellMatPoly<Dim> >(cell_id_);
105 
106  //MatPoly corresponding to a non-convex cell can have non-planar faces
107  MatPoly<Dim> cell_matpoly;
108  if (convex_cell_)
109  cell_matpoly = reconstructor_.cell_matpoly(cell_id_);
110  else
111  reconstructor_.cell_matpoly(cell_id_).faceted_matpoly(&cell_matpoly);
112 
113  double vol_tol = reconstructor_.iterative_methods_tolerances()[0].fun_eps;
114  double dst_tol = reconstructor_.iterative_methods_tolerances()[0].arg_eps;
115 
116  Plane_t<Dim> cutting_plane;
117  HalfSpaceSets_t<Dim> hs_sets; // Structure containing vectors of MatPoly's and their
118  // aggregated moments for the respective half-spaces
119 
120  //Original vector of mixed MatPoly's consists of the cell's MatPoly
121  hs_sets.upper_halfspace_set.matpolys = {cell_matpoly};
122 
123  //Create Splitter instance: assumes split MatPoly's are convex
124  MatPoly_Splitter split_matpolys(hs_sets.upper_halfspace_set.matpolys,
125  cutting_plane, vol_tol, dst_tol, true);
126 
127  for (int imat = 0; imat < nmats; imat++) {
128  int matid = mat_ids[cell_materials_order_[permutation_ID][imat]];
129 
130  MatPolySet_t<Dim>* single_mat_set_ptr;
131  //On the last iteration the remaining part is single-material,
132  //so we don't need to split it
133  if (imat == nmats - 1)
134  single_mat_set_ptr = &hs_sets.upper_halfspace_set;
135  else {
136  // Find cutting plane position: assumes all faces are planar
137  // Note that this method generally does NOT require MatPoly_Splitter,
138  // but uses MatPoly_Clipper instead. Nested dissections itself
139  // has neither MatPoly_Clipper, nor MeshWrapper, those are
140  // handled by the reconstructor that created this nested dissections
141  // instance.
142  reconstructor_.get_plane_position(cell_id_, matid,
143  hs_sets.upper_halfspace_set.matpolys,
144  cutting_plane, true);
145 
146  // On the first cut we have to decompose a non-convex MatPoly
147  // This is done inside the loop to make the first step cheaper:
148  // finding the position of the cutting plane normally requires
149  // only computation of moments, which we assume to be possible
150  // without decomposing into tetrahedrons (e.g. if r3d is used).
151  if (!convex_cell_ && (imat == 0)) {
152  hs_sets.upper_halfspace_set.matpolys.clear();
153  cell_matpoly.decompose(hs_sets.upper_halfspace_set.matpolys);
154  }
155 
156  hs_sets = split_matpolys();
157  //Single-material MatPoly's are below the plane
158  single_mat_set_ptr = &hs_sets.lower_halfspace_set;
159 
160  // Filter out MatPoly's with volumes below tolerance
161  int ismp = 0;
162  while (ismp < single_mat_set_ptr->matpolys.size())
163  if (single_mat_set_ptr->matpolys[ismp].moments()[0] >= vol_tol)
164  ismp++;
165  else {
166  for (int im = 0; im < Dim + 1; im++)
167  single_mat_set_ptr->moments[im] -=
168  single_mat_set_ptr->matpolys[ismp].moments()[im];
169 
170  single_mat_set_ptr->matpolys.erase(
171  single_mat_set_ptr->matpolys.begin() + ismp);
172  }
173  }
174 
175  // Add single-material poly's below the plane to CellMatPoly
176  for (int ismp = 0; ismp < single_mat_set_ptr->matpolys.size(); ismp++) {
177  MatPoly<Dim>& cur_matpoly = single_mat_set_ptr->matpolys[ismp];
178  cur_matpoly.set_mat_id(matid);
179  cmp_ptr->add_matpoly(cur_matpoly);
180  }
181 
182  if (single_mat_set_ptr->matpolys.size() > 0)
183  cmp_ptr->assign_material_moments(matid, single_mat_set_ptr->moments);
184  }
185 
186  return cmp_ptr;
187  }
